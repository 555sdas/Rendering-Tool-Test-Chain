### 2.2 核心管理器 (Runtime/Core/XRTestManager.cs)

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using UnityEngine;
using UnityEngine.XR;

namespace XRTestCollector.Core
{
    public class XRTestManager : MonoBehaviour
    {
        private static XRTestManager _instance;
        public static XRTestManager Instance
        {
            get
            {
                if (_instance == null)
                {
                    var go = new GameObject("XRTestManager");
                    _instance = go.AddComponent<XRTestManager>();
                    DontDestroyOnLoad(go);
                }
                return _instance;
            }
        }

        [Header("采集配置")]
        [SerializeField] private float sampleInterval = 0.1f; // 100ms采样间隔
        [SerializeField] private bool autoStartOnPlay = false;
        [SerializeField] private bool captureScreenshots = false;
        [SerializeField] private int screenshotInterval = 60; // 每秒截图帧数

        private XRTestSession _currentSession;
        private List<IPerformanceCollector> _collectors;
        private float _sampleTimer;
        private int _frameCount;
        private float _fpsTimer;
        private float _currentFps;
        private bool _isRecording;

        // 事件
        public event Action<PerformanceSample> OnSampleRecorded;
        public event Action<XRTestSession> OnSessionStarted;
        public event Action<XRTestSession> OnSessionStopped;
        public event Action<string> OnError;

        void Awake()
        {
            if (_instance != null && _instance != this)
            {
                Destroy(gameObject);
                return;
            }
            _instance = this;
            DontDestroyOnLoad(gameObject);
            
            InitializeCollectors();
        }

        void Start()
        {
            if (autoStartOnPlay)
            {
                StartSession();
            }
        }

        void Update()
        {
            if (!_isRecording) return;

            // FPS计算
            _frameCount++;
            _fpsTimer += Time.unscaledDeltaTime;
            if (_fpsTimer >= 1.0f)
            {
                _currentFps = _frameCount / _fpsTimer;
                _frameCount = 0;
                _fpsTimer = 0;
            }

            // 定时采样
            _sampleTimer += Time.unscaledDeltaTime;
            if (_sampleTimer >= sampleInterval)
            {
                RecordSample();
                _sampleTimer = 0;
            }

            // 截图
            if (captureScreenshots && _frameCount % screenshotInterval == 0)
            {
                CaptureScreenshot();
            }
        }

        private void InitializeCollectors()
        {
            _collectors = new List<IPerformanceCollector>
            {
                new FrameTimeCollector(),
                new CpuUsageCollector(),
                new GpuUsageCollector(),
                new MemoryCollector(),
                new VramCollector()
            };
        }

        public void StartSession(XRTestConfig config = null)
        {
            if (_isRecording)
            {
                Debug.LogWarning("[XRTestManager] 已有会话正在运行");
                return;
            }

            try
            {
                _currentSession = new XRTestSession
                {
                    SessionId = Guid.NewGuid().ToString(),
                    StartTime = DateTime.UtcNow,
                    Config = config ?? new XRTestConfig(),
                    DeviceInfo = DeviceInfoCollector.Collect(),
                    SceneInfo = SceneInfoCollector.Collect()
                };

                foreach (var collector in _collectors)
                {
                    collector.Start();
                }

                _isRecording = true;
                _sampleTimer = 0;
                _frameCount = 0;
                _fpsTimer = 0;

                OnSessionStarted?.Invoke(_currentSession);
                Debug.Log($"[XRTestManager] 测试会话已启动: {_currentSession.SessionId}");
            }
            catch (Exception ex)
            {
                OnError?.Invoke($"启动会话失败: {ex.Message}");
            }
        }

        public void StopSession()
        {
            if (!_isRecording) return;

            _isRecording = false;
            _currentSession.EndTime = DateTime.UtcNow;

            foreach (var collector in _collectors)
            {
                collector.Stop();
            }

            OnSessionStopped?.Invoke(_currentSession);
            Debug.Log($"[XRTestManager] 测试会话已结束: {_currentSession.SessionId}, " +
                      $"持续时间: {_currentSession.Duration.TotalSeconds:F2}s, " +
                      $"采样数: {_currentSession.Samples.Count}");
        }

        private void RecordSample()
        {
            var sample = new PerformanceSample
            {
                Timestamp = DateTime.UtcNow,
                FrameIndex = Time.frameCount,
                Fps = _currentFps,
                DeltaTime = Time.unscaledDeltaTime * 1000 // ms
            };

            // 采集各指标
            foreach (var collector in _collectors)
            {
                try
                {
                    collector.Collect(ref sample);
                }
                catch (Exception ex)
                {
                    Debug.LogWarning($"[XRTestManager] 采集器 {collector.GetType().Name} 失败: {ex.Message}");
                }
            }

            _currentSession.Samples.Add(sample);
            OnSampleRecorded?.Invoke(sample);
        }

        private void CaptureScreenshot()
        {
            var filename = $"screenshot_{_currentSession.SessionId}_{Time.frameCount}.png";
            var path = System.IO.Path.Combine(_currentSession.Config.ScreenshotPath, filename);
            ScreenCapture.CaptureScreenshot(path);
            _currentSession.ScreenshotPaths.Add(path);
        }

        public XRTestSession GetCurrentSession()
        {
            return _currentSession;
        }

        public bool IsRecording => _isRecording;

        void OnDestroy()
        {
            if (_isRecording)
            {
                StopSession();
            }
        }

        void OnApplicationQuit()
        {
            if (_isRecording)
            {
                StopSession();
            }
        }
    }
}
```

                GraphicsDeviceVersion = SystemInfo.graphicsDeviceVersion,
                GraphicsMemorySize = SystemInfo.graphicsMemorySize,
                GraphicsShaderLevel = SystemInfo.graphicsShaderLevel,
                
                // XR信息
                XrLoaded = XRSettings.loaded,
                XrEnabled = XRSettings.enabled,
                XrDeviceName = XRSettings.loaded ? XRSettings.supportedDevices.FirstOrDefault() : "None",
                
                // 渲染配置
                RenderPipeline = GraphicsSettings.defaultRenderPipeline?.GetType().Name ?? "Built-in",
                ColorSpace = QualitySettings.activeColorSpace.ToString(),
                TargetFrameRate = Application.targetFrameRate,
                VSyncCount = QualitySettings.vSyncCount,
                
                // 分辨率
                ScreenWidth = Screen.width,
                ScreenHeight = Screen.height,
                ScreenDpi = Screen.dpi,
                
                // Unity版本
                UnityVersion = Application.unityVersion,
                
                // 采集时间
                CollectionTime = System.DateTime.UtcNow
            };

            // XR特定信息
            if (XRSettings.loaded)
            {
                var headDevices = new List<InputDevice>();
                InputDevices.GetDevicesAtXRNode(XRNode.Head, headDevices);
                if (headDevices.Count > 0)
                {
                    var headDevice = headDevices[0];
                    info.XrDeviceManufacturer = headDevice.manufacturer;
                    info.XrDeviceSerialNumber = headDevice.serialNumber;
                }
            }

            return info;
        }
    }
}

namespace XRTestCollector.Exporters

    public class JsonExporter : IDataExporter
    {
        private readonly JsonSerializerOptions _options;

        public JsonExporter()
        {
            _options = new JsonSerializerOptions
            {
                WriteIndented = true,
                PropertyNamingPolicy = JsonNamingPolicy.CamelCase,
                DefaultIgnoreCondition = JsonIgnoreCondition.WhenWritingNull
            };
        }

        public void Export(XRTestSession session, string outputPath)
        {
            var exportData = new TestExportData
            {
                ExportVersion = "1.0",
                ExportTime = DateTime.UtcNow,
                Session = session
            };

            var json = JsonSerializer.Serialize(exportData, _options);

            Directory.CreateDirectory(outputPath);
            File.WriteAllText(fullPath, json);

        }

        public string GetFileExtension() => ".json";
    }

    session_id: int,

    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)

    check_permission(current_user.role, Permission.TEST_EXECUTE)

        raise HTTPException(status_code=400, detail="仅支持JSON格式文件")

    data = json.loads(content)

    return {"message": "文件上传成功，正在处理", "session_id": session_id}

### 3.2 数据采集服务 (backend/app/services/data_collection_service.py)

```python
from sqlalchemy.orm import Session
from sqlalchemy import func
from typing import List, Optional, Dict, Any
from datetime import datetime
import json

from app.models.test_session import TestSession, TestSessionStatus
from app.models.performance_sample import PerformanceSample
from app.models.scene_asset import SceneAsset

class DataCollectionService:
    def __init__(self, db: Session):
        self.db = db
    
    def create_session(
        self,
        project_id: int,
        user_id: int,
        scene_id: Optional[int] = None,
        device_info: Optional[Dict] = None,
        runtime_config: Optional[Dict] = None
    ) -> TestSession:
        session = TestSession(
            project_id=project_id,
            user_id=user_id,
            scene_id=scene_id,
            device_info=device_info or {},
            runtime_config=runtime_config or {},
            status=TestSessionStatus.RUNNING,
            start_time=datetime.utcnow()
        )
        self.db.add(session)
        self.db.commit()
        self.db.refresh(session)
        return session
    
    def add_samples_batch(self, session_id: int, samples: List[Dict]) -> int:
        db_samples = []
        for sample_data in samples:
            sample = PerformanceSample(
                session_id=session_id,
                timestamp=datetime.fromisoformat(sample_data.get('timestamp', datetime.utcnow().isoformat())),
                frame_index=sample_data.get('frameIndex', 0),
                fps=sample_data.get('fps', 0),
                delta_time_ms=sample_data.get('deltaTime', 0),
                cpu_usage_percent=sample_data.get('cpuUsagePercent'),
                cpu_time_ms=sample_data.get('cpuTimeMs'),
                gpu_usage_percent=sample_data.get('gpuUsagePercent'),
                gpu_time_ms=sample_data.get('gpuTimeMs'),
                memory_usage_bytes=sample_data.get('memoryUsageBytes'),
                vram_usage_bytes=sample_data.get('vramUsageBytes'),
                draw_calls=sample_data.get('drawCalls'),
                set_pass_calls=sample_data.get('setPassCalls'),
                triangles=sample_data.get('triangles'),
                vertices=sample_data.get('vertices'),
                xr_render_time_ms=sample_data.get('xrRenderTimeMs'),
                xr_frame_dropped=sample_data.get('xrFrameDropped', False)
            )
            db_samples.append(sample)
        
        self.db.bulk_save_objects(db_samples)
        self.db.commit()
        return len(db_samples)
    
    def process_uploaded_session(self, session_id: int, data: Dict):
        session = self.db.query(TestSession).filter(TestSession.id == session_id).first()
        if not session:
            return
        
        # 更新设备信息
        if 'deviceInfo' in data:
            session.device_info = data['deviceInfo']
        
        # 更新场景信息
        if 'sceneInfo' in data:
            session.scene_info = data['sceneInfo']
        
        # 处理采样数据
        if 'samples' in data:
            self.add_samples_batch(session_id, data['samples'])
        
        self.db.commit()
    
    def complete_session(self, session_id: int) -> TestSession:
        session = self.db.query(TestSession).filter(TestSession.id == session_id).first()
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        session.status = TestSessionStatus.COMPLETED
        session.end_time = datetime.utcnow()
        
        # 计算统计指标
        stats = self.db.query(
            func.avg(PerformanceSample.fps).label('avg_fps'),
            func.min(PerformanceSample.fps).label('min_fps'),
            func.max(PerformanceSample.fps).label('max_fps'),
            func.avg(PerformanceSample.delta_time_ms).label('avg_frame_time'),
            func.avg(PerformanceSample.cpu_usage_percent).label('avg_cpu'),
            func.avg(PerformanceSample.gpu_usage_percent).label('avg_gpu'),
            func.max(PerformanceSample.memory_usage_bytes).label('peak_memory'),
            func.count().label('sample_count')
        ).filter(PerformanceSample.session_id == session_id).first()
        
        session.summary_stats = {
            "avg_fps": round(stats.avg_fps, 2) if stats.avg_fps else None,
            "min_fps": round(stats.min_fps, 2) if stats.min_fps else None,
            "max_fps": round(stats.max_fps, 2) if stats.max_fps else None,
            "avg_frame_time_ms": round(stats.avg_frame_time, 2) if stats.avg_frame_time else None,
            "avg_cpu_percent": round(stats.avg_cpu, 2) if stats.avg_cpu else None,
            "avg_gpu_percent": round(stats.avg_gpu, 2) if stats.avg_gpu else None,
            "peak_memory_mb": round(stats.peak_memory / (1024 * 1024), 2) if stats.peak_memory else None,
            "sample_count": stats.sample_count
        }
        
        self.db.commit()
        self.db.refresh(session)
        return session
    
    def get_samples(self, session_id: int, skip: int = 0, limit: int = 1000) -> List[Dict]:
        samples = self.db.query(PerformanceSample).filter(
            PerformanceSample.session_id == session_id
        ).order_by(PerformanceSample.timestamp).offset(skip).limit(limit).all()
        
        return [{
            "timestamp": s.timestamp.isoformat(),
            "frame_index": s.frame_index,
            "fps": s.fps,
            "delta_time_ms": s.delta_time_ms,
            "cpu_usage_percent": s.cpu_usage_percent,
            "gpu_usage_percent": s.gpu_usage_percent,
            "memory_usage_mb": round(s.memory_usage_bytes / (1024 * 1024), 2) if s.memory_usage_bytes else None,
            "vram_usage_mb": round(s.vram_usage_bytes / (1024 * 1024), 2) if s.vram_usage_bytes else None,
        } for s in samples]
```

class TestTask(Base):

    # 执行配置

### 5.2 数据导出服务 (backend/app/services/export_service.py)

```python
import csv
import json
import io
from typing import List, Dict, Any
from datetime import datetime

class ExportService:
    @staticmethod
    def export_to_csv(data: List[Dict], columns: List[str]) -> str:
        output = io.StringIO()
        writer = csv.DictWriter(output, fieldnames=columns)
        writer.writeheader()
        writer.writerows(data)
        return output.getvalue()
    
    @staticmethod
    def export_to_json(data: List[Dict], pretty: bool = True) -> str:
        if pretty:
            return json.dumps(data, ensure_ascii=False, indent=2, default=str)
        return json.dumps(data, ensure_ascii=False, default=str)
    
    @staticmethod
    def export_to_excel(data: List[Dict], sheet_name: str = "Sheet1") -> bytes:
        import openpyxl
        from openpyxl.styles import Font, Alignment
        
        wb = openpyxl.Workbook()
        ws = wb.active
        ws.title = sheet_name
        
        if not data:
            return b''
        
        # 写入表头
        headers = list(data[0].keys())
        ws.append(headers)
        
        # 设置表头样式
        for cell in ws[1]:
            cell.font = Font(bold=True)
            cell.alignment = Alignment(horizontal='center')
        
        # 写入数据
        for row in data:
            ws.append([row.get(h) for h in headers])
        
        # 自动调整列宽
        for column in ws.columns:
            max_length = 0
            column_letter = column[0].column_letter
            for cell in column:
                try:
                    if cell.value:
                        max_length = max(max_length, len(str(cell.value)))
                except:
                    pass
            adjusted_width = min(max_length + 2, 50)
            ws.column_dimensions[column_letter].width = adjusted_width
        
        output = io.BytesIO()
        wb.save(output)
        output.seek(0)
        return output.getvalue()
    
    @staticmethod
    def export_session_data(session_data: Dict, format_type: str = "json") -> Any:
        samples = session_data.get('samples', [])
        
        if format_type == "csv":
            if samples:
                columns = list(samples[0].keys())
                return ExportService.export_to_csv(samples, columns)
            return ""
        
        elif format_type == "json":
            return ExportService.export_to_json(session_data)
        
        elif format_type == "excel":
            return ExportService.export_to_excel(samples, "Performance Data")
        
        else:
            raise ValueError(f"Unsupported export format: {format_type}")
```
