# 阶段五：场景库与终端适配

## 1. 目标

构建可复用的标准测试场景库和3D模型用例库，实现多终端适配能力，包括：
- 标准测试场景库（≥6个，覆盖低/中/高复杂度）
- 3D模型用例库（光照/阴影/透明/粒子/反射/动态物体/物理仿真）
- 多终端适配层（≥2类XR终端）
- 端云渲染/串流渲染基础测试
- 国内外工具链接入能力

### 3.2 终端适配服务 (backend/app/services/device_adapter_service.py)

```python
from typing import Dict, List, Optional, Any
from sqlalchemy.orm import Session

from app.models.device_profile import DeviceProfile, DeviceType
from app.models.test_session import TestSession

class DeviceAdapterService:
    def __init__(self, db: Session):
        self.db = db
    
    def register_device(self, device_info: Dict) -> DeviceProfile:
        device_id = device_info.get("deviceId") or self._generate_device_id(device_info)
        
        existing = self.db.query(DeviceProfile).filter(
            DeviceProfile.device_id == device_id
        ).first()
        
        if existing:
            # 更新现有设备信息
            existing.hardware_specs = device_info.get("hardware", {})
            existing.graphics_capabilities = device_info.get("graphics", {})
            existing.available_metrics = self._detect_available_metrics(device_info)
            self.db.commit()
            return existing
        
        device = DeviceProfile(
            device_id=device_id,
            name=device_info.get("name", "Unknown Device"),
            manufacturer=device_info.get("manufacturer", "Unknown"),
            device_type=self._classify_device_type(device_info),
            hardware_specs=device_info.get("hardware", {}),
            supported_platforms=device_info.get("platforms", []),
            graphics_capabilities=device_info.get("graphics", {}),
            sensor_capabilities=device_info.get("sensors", {}),
            available_metrics=self._detect_available_metrics(device_info),
            metric_limitations=self._generate_limitations(device_info)
        )
        
        self.db.add(device)
        self.db.commit()
        self.db.refresh(device)
        return device
    
    def _generate_device_id(self, device_info: Dict) -> str:
        import hashlib
        data = f"{device_info.get('name', '')}-{device_info.get('manufacturer', '')}"
        return hashlib.md5(data.encode()).hexdigest()[:12]
    
    def _classify_device_type(self, device_info: Dict) -> DeviceType:
        name = device_info.get("name", "").lower()
        
        if "quest" in name or "pico" in name or "htc" in name:
            return DeviceType.VR_HEADSET
        elif "hololens" in name or "magic" in name:
            return DeviceType.AR_GLASSES
        elif "iphone" in name or "android" in name:
            return DeviceType.MOBILE_AR
        elif "simulator" in name or "editor" in name:
            return DeviceType.PC_SIMULATOR
        
        return DeviceType.VR_HEADSET
    
    def _detect_available_metrics(self, device_info: Dict) -> List[str]:
        metrics = ["fps", "frame_time", "memory_usage"]
        
        hardware = device_info.get("hardware", {})
        
        # CPU指标
        if hardware.get("cpuCores"):
            metrics.append("cpu_usage")
        
        # GPU指标
        if hardware.get("gpuName"):
            metrics.append("gpu_usage")
            metrics.append("gpu_time")
        
        # 显存
        if hardware.get("vramSizeMB"):
            metrics.append("vram_usage")
        
        # 温度
        if hardware.get("supportsTemperature"):
            metrics.append("temperature")
        
        # 功耗
        if hardware.get("supportsPowerConsumption"):
            metrics.append("power_consumption")
        
        return metrics
    
    def _generate_limitations(self, device_info: Dict) -> str:
        limitations = []
        hardware = device_info.get("hardware", {})
        
        if not hardware.get("gpuName"):
            limitations.append("无法直接采集GPU利用率，需使用间接方法估算")
        
        if not hardware.get("supportsTemperature"):
            limitations.append("不支持温度采集")
        
        if not hardware.get("supportsPowerConsumption"):
            limitations.append("不支持功耗采集")
        
        return "; ".join(limitations) if limitations else "无已知限制"
    
    def get_device_capabilities(self, device_id: str) -> Dict:
        device = self.db.query(DeviceProfile).filter(
            DeviceProfile.device_id == device_id
        ).first()
        
        if not device:
            raise ValueError(f"Device {device_id} not found")
        
        return {
            "device_id": device.device_id,
            "name": device.name,
            "type": device.device_type.value,
            "available_metrics": device.available_metrics,
            "limitations": device.metric_limitations,
            "graphics_capabilities": device.graphics_capabilities,
            "sensor_capabilities": device.sensor_capabilities
        }
    
    def get_compatible_scenes(self, device_id: str) -> List[Dict]:
        device = self.db.query(DeviceProfile).filter(
            DeviceProfile.device_id == device_id
        ).first()
        
        if not device:
            raise ValueError(f"Device {device_id} not found")
        
        # 根据设备能力筛选兼容场景
        from app.models.scene_asset import SceneAsset, SceneAssetStatus
        
        scenes = self.db.query(SceneAsset).filter(
            SceneAsset.status == SceneAssetStatus.ACTIVE
        ).all()
        
        compatible = []
        for scene in scenes:
            # 检查平台兼容性
            target_platforms = scene.target_platforms or []
            device_platforms = device.supported_platforms or []
            
            if any(p in device_platforms for p in target_platforms):
                compatible.append({
                    "scene_id": scene.scene_id,
                    "name": scene.name,
                    "complexity": scene.complexity_level.value,
                    "performance_targets": scene.performance_targets
                })
        
        return compatible
```

from typing import Dict, List, Optional, Protocol
from abc import ABC, abstractmethod

class ToolchainAdapter(ABC):
    @abstractmethod
    def get_name(self) -> str:
        pass
    
    @abstractmethod
    def get_version(self) -> str:
        pass
    
    @abstractmethod
    def get_capabilities(self) -> List[str]:
        pass
    
    @abstractmethod
    def collect_data(self, config: Dict) -> Dict:
        pass
    
    @abstractmethod
    def get_supported_metrics(self) -> List[str]:
        pass

class UnityAdapter(ToolchainAdapter):
    def get_name(self) -> str:
        return "Unity"

            "gpu_usage",

            "profiler_data"

        return [
            "frame_rate",
            "frame_time",
            "cpu_usage",
            "memory_usage",
            "draw_calls"
        ]
    
    def collect_data(self, config: Dict) -> Dict:
        return {
            "toolchain": "Cocos Creator",
            "data": {}
        }
    
    def get_supported_metrics(self) -> List[str]:
        return self.get_capabilities()

class ToolchainManager:
    def __init__(self):
        self.adapters: Dict[str, ToolchainAdapter] = {
            "unity": UnityAdapter(),
            "unreal": UnrealAdapter(),
            "cocos": CocosAdapter()
        }
    
    def get_adapter(self, name: str) -> Optional[ToolchainAdapter]:
        return self.adapters.get(name.lower())
    
    def list_adapters(self) -> List[Dict]:
        return [
            {
                "name": adapter.get_name(),
                "version": adapter.get_version(),
                "capabilities": adapter.get_capabilities(),
                "supported_metrics": adapter.get_supported_metrics()
            }
            for adapter in self.adapters.values()
        ]
    
    def register_adapter(self, name: str, adapter: ToolchainAdapter):
        self.adapters[name.lower()] = adapter

// frontend/src/pages/Scenes/SceneLibrary.tsx
import React, { useState, useEffect } from 'react';
import { Card, Row, Col, Tag, Button, Modal, Descriptions, List, message } from 'antd';
import { apiClient } from '../../api/client';

interface SceneAsset {
  scene_id: string;
  name: string;
  complexity_level: string;
  technical_features: string[];
  application_category: string;
  description: string;
  performance_targets: Record<string, any>;
  test_points: Array<Record<string, any>>;
}

export const SceneLibrary: React.FC = () => {
  const [scenes, setScenes] = useState<SceneAsset[]>([]);
  const [selectedScene, setSelectedScene] = useState<SceneAsset | null>(null);
  const [modalVisible, setModalVisible] = useState(false);

  useEffect(() => {
    fetchScenes();
  }, []);

  const fetchScenes = async () => {
    try {
      const response = await apiClient.get('/scenes');
      setScenes(response.data.scenes);
    } catch (error) {
      message.error('获取场景库失败');
    }
  };

  const getComplexityColor = (level: string) => {
    const colors: Record<string, string> = {
      low: 'green',
      medium: 'orange',
      high: 'red'
    };
    return colors[level] || 'default';
  };

  return (
    <div>
      <h2>标准测试场景库</h2>
      <Row gutter={[16, 16]}>
        {scenes.map(scene => (
          <Col span={8} key={scene.scene_id}>
            <Card
              title={scene.name}
              extra={
                <Tag color={getComplexityColor(scene.complexity_level)}>
                  {scene.complexity_level}
                </Tag>
              }
              actions={[
                <Button type="link" onClick={() => {
                  setSelectedScene(scene);
                  setModalVisible(true);
                }}>详情</Button>,
                <Button type="link">测试</Button>
              ]}
            >
              <p>{scene.description}</p>
              <div>
                {scene.technical_features.map(feature => (
                  <Tag key={feature}>{feature}</Tag>
                ))}
              </div>
            </Card>
          </Col>
        ))}
      </Row>

      <Modal
        title={selectedScene?.name}
        visible={modalVisible}
        onCancel={() => setModalVisible(false)}
        footer={null}
        width={800}
      >
        {selectedScene && (
          <Descriptions bordered column={2}>
            <Descriptions.Item label="场景ID">{selectedScene.scene_id}</Descriptions.Item>
            <Descriptions.Item label="复杂度">
              <Tag color={getComplexityColor(selectedScene.complexity_level)}>
                {selectedScene.complexity_level}
              </Tag>
            </Descriptions.Item>
            <Descriptions.Item label="应用场景">{selectedScene.application_category}</Descriptions.Item>
            <Descriptions.Item label="技术特性">
              {selectedScene.technical_features.join(', ')}
            </Descriptions.Item>
            <Descriptions.Item label="性能目标" span={2}>
              <pre>{JSON.stringify(selectedScene.performance_targets, null, 2)}</pre>
            </Descriptions.Item>
            <Descriptions.Item label="测试点" span={2}>
              <List
                size="small"
                dataSource={selectedScene.test_points}
                renderItem={item => (
                  <List.Item>
                    <List.Item.Meta
                      title={item.name}
                      description={item.description}
                    />
                  </List.Item>
                )}
              />
            </Descriptions.Item>
          </Descriptions>
        )}
      </Modal>
    </div>
  );
};

## 7. 验收标准

| 验收项 | 标准 |
|--------|------|
| 场景数量 | 交付不少于6个标准测试场景 |
| 复杂度覆盖 | 覆盖低/中/高三种复杂度 |
| 技术特性 | 覆盖光照/阴影/透明/粒子/反射/动态物体/物理仿真 |

| 终端适配 | 支持不少于2类XR终端 |
