version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    environment:
      POSTGRES_DB: xr_test_db
      POSTGRES_USER: xruser
      POSTGRES_PASSWORD: ${DB_PASSWORD:-xrpass}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./init-scripts:/docker-entrypoint-initdb.d
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U xruser -d xr_test_db"]
      interval: 5s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 5s
      retries: 5

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    environment:
      DATABASE_URL: postgresql://xruser:${DB_PASSWORD:-xrpass}@postgres:5432/xr_test_db
      REDIS_URL: redis://redis:6379/0
      SECRET_KEY: ${SECRET_KEY:-change-me-in-production}
      ENCRYPTION_MASTER_KEY: ${ENCRYPTION_MASTER_KEY:-change-me}
    ports:
      - "8000:8000"
    volumes:
      - uploads:/app/uploads
      - backups:/app/backups
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    ports:
      - "80:80"
    depends_on:
      - backend

  nginx:
    image: nginx:alpine
    ports:
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
      - ./ssl:/etc/nginx/ssl:ro
    depends_on:
      - frontend
      - backend

volumes:
  postgres_data:
  redis_data:
  uploads:
  backups:

## 6. 许可证汇总

### 6.1 许可证分布

| 许可证类型 | 数量 | 说明 |
|------------|------|------|
| MIT | 25+ | 最宽松的开源许可证 |
| BSD/BSD-3-Clause | 10+ | 宽松的开源许可证 |
| Apache-2.0 | 8+ | 允许商业使用 |
| LGPL | 1 | 弱 copyleft |
| HPND | 1 | 历史许可证 |
| CC0 | 1 | 公共领域 |

### 6.2 合规性说明

1. **MIT许可证**: 允许自由使用、修改、分发，包括商业用途
2. **BSD许可证**: 类似MIT，允许商业使用，需保留版权声明
3. **Apache-2.0**: 允许商业使用，提供专利授权保护
4. **LGPL**: 允许链接到闭源软件，修改部分需开源
5. **HPND**: Pillow使用的历史许可证，允许商业使用

### 6.3 商用合规建议

- 所有依赖均为商业友好型许可证
- 无需开源自身代码（除非修改LGPL组件）
- 需在文档中保留第三方组件的版权声明
- 建议定期使用工具扫描依赖许可证变化

## 7. 版本锁定策略

### 7.1 版本管理原则

1. **精确版本**: 生产环境使用精确版本号（==）
2. **锁定文件**: 使用 poetry.lock / package-lock.json 锁定完整依赖树
3. **定期更新**: 每月检查安全更新，每季度评估功能更新
4. **兼容性测试**: 更新依赖前在测试环境验证

### 7.2 依赖更新流程

```
1. 检查更新
   pip list --outdated
   npm outdated

2. 评估影响
   - 查看CHANGELOG
   - 检查破坏性变更
   - 评估安全修复

3. 更新测试环境
   pip install --upgrade <package>
   npm update <package>

4. 执行测试
   pytest
   npm test

5. 验证功能
   - 核心功能回归测试
   - 性能基准测试

6. 更新生产环境
   - 更新 requirements.txt / package.json
   - 更新锁定文件
   - 部署到生产
```
