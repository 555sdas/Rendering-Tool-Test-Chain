# 部署与迁移文档

## 1. 环境迁移指南

### 1.1 从开发环境迁移到测试环境

```bash
# 1. 在开发环境打包代码
cd backend
git archive -o backend.tar.gz HEAD

cd ../frontend
npm run build
tar -czf frontend-dist.tar.gz dist/

# 2. 复制到测试服务器
scp backend.tar.gz user@test-server:/opt/xr-test/
scp frontend-dist.tar.gz user@test-server:/opt/xr-test/
scp docker-compose.yml user@test-server:/opt/xr-test/
scp .env.example user@test-server:/opt/xr-test/.env

# 3. 在测试服务器部署
ssh user@test-server
cd /opt/xr-test

# 解压后端代码
tar -xzf backend.tar.gz -C backend/

# 配置环境变量
cp .env.example .env
vim .env  # 修改数据库地址、密钥等

# 启动服务
docker-compose up -d

# 执行数据库迁移
docker-compose exec backend alembic upgrade head
```

### 1.2 从测试环境迁移到生产环境

```bash
# 1. 备份测试环境数据
docker-compose exec postgres pg_dump -U xruser xr_test_db > test_backup.sql

# 2. 复制到生产服务器
scp test_backup.sql user@prod-server:/opt/xr-test/
scp -r backend/ user@prod-server:/opt/xr-test/
scp -r frontend/dist/ user@prod-server:/opt/xr-test/frontend/

# 3. 在生产环境部署
ssh user@prod-server
cd /opt/xr-test

# 修改生产环境配置
vim .env
# 修改以下配置:
# DEBUG=false
# SECRET_KEY=强密码
# DATABASE_URL=生产数据库地址
# 启用HTTPS

# 启动服务
docker-compose -f docker-compose.yml -f docker-compose.prod.yml up -d

# 恢复数据
docker-compose exec -T postgres psql -U xruser xr_test_db < test_backup.sql

# 验证部署
curl https://your-domain.com/api/v1/health
```

### 1.3 跨平台迁移（Windows → Linux）

```bash
# Windows环境准备
# 1. 确保使用WSL2或Git Bash执行以下命令
# 2. 安装Docker Desktop并启用WSL2后端

# 导出数据卷
docker run --rm -v xr-test-platform_postgres_data:/data -v $(pwd):/backup alpine tar czf /backup/postgres_data.tar.gz -C /data .
docker run --rm -v xr-test-platform_uploads:/data -v $(pwd):/backup alpine tar czf /backup/uploads.tar.gz -C /data .

# 复制到Linux服务器
scp postgres_data.tar.gz user@linux-server:/opt/xr-test/
scp uploads.tar.gz user@linux-server:/opt/xr-test/

# Linux服务器恢复
cd /opt/xr-test

# 创建数据卷并恢复
docker volume create xr-test-platform_postgres_data
docker volume create xr-test-platform_uploads

docker run --rm -v xr-test-platform_postgres_data:/data -v $(pwd):/backup alpine tar xzf /backup/postgres_data.tar.gz -C /data
docker run --rm -v xr-test-platform_uploads:/data -v $(pwd):/backup alpine tar xzf /backup/uploads.tar.gz -C /data

# 启动服务
docker-compose up -d
```

## 2. Docker Compose配置详解

### 2.1 开发环境配置

```yaml
# docker-compose.yml
version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    environment:
      POSTGRES_DB: ${POSTGRES_DB:-xr_test_db}
      POSTGRES_USER: ${POSTGRES_USER:-xruser}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-xrpass}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./init-scripts:/docker-entrypoint-initdb.d
    ports:
      - "${POSTGRES_PORT:-5432}:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER:-xruser} -d ${POSTGRES_DB:-xr_test_db}"]
      interval: 5s
      timeout: 5s
      retries: 5
    restart: unless-stopped

  redis:
    image: redis:7-alpine
    ports:
      - "${REDIS_PORT:-6379}:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 5s
      retries: 5
    restart: unless-stopped

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    environment:
      DATABASE_URL: postgresql://${POSTGRES_USER:-xruser}:${POSTGRES_PASSWORD:-xrpass}@postgres:5432/${POSTGRES_DB:-xr_test_db}
      REDIS_URL: redis://redis:6379/0
      SECRET_KEY: ${SECRET_KEY:-change-me-in-production}
      DEBUG: ${DEBUG:-false}
      ACCESS_TOKEN_EXPIRE_MINUTES: ${ACCESS_TOKEN_EXPIRE_MINUTES:-30}
      REFRESH_TOKEN_EXPIRE_DAYS: ${REFRESH_TOKEN_EXPIRE_DAYS:-7}
    ports:
      - "${BACKEND_PORT:-8000}:8000"
    volumes:
      - uploads:/app/uploads
      - backups:/app/backups
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/api/v1/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    ports:
      - "${NGINX_PORT:-80}:80"
    depends_on:
      - backend
    restart: unless-stopped

volumes:
  postgres_data:
  redis_data:
  uploads:
  backups:
```

## 3. 数据库迁移策略

### 3.1 Alembic迁移命令

```bash
# 进入后端容器
docker-compose exec backend bash

# 创建自动迁移脚本
alembic revision --autogenerate -m "add_new_table"

# 查看迁移历史
alembic history --verbose

# 升级到最新版本
alembic upgrade head

# 降级到指定版本
alembic downgrade <revision_id>

# 降级到上一个版本
alembic downgrade -1

# 查看当前版本
alembic current

# 标记数据库为指定版本（不执行迁移）
alembic stamp <revision_id>
```

### 3.2 数据迁移脚本示例

```python
# backend/alembic/versions/xxxx_add_user_preferences.py
"""Add user preferences table

Revision ID: xxxx
Revises: yyyy
Create Date: 2024-01-01 00:00:00.000000

"""
from alembic import op
import sqlalchemy as sa
from sqlalchemy.dialects import postgresql

# revision identifiers, used by Alembic.
revision = 'xxxx'
down_revision = 'yyyy'
branch_labels = None
depends_on = None


def upgrade():
    # 创建新表
    op.create_table(
        'user_preferences',
        sa.Column('id', sa.Integer(), nullable=False),
        sa.Column('user_id', sa.Integer(), nullable=False),
        sa.Column('preferences', postgresql.JSON(astext_type=sa.Text()), nullable=True),
        sa.Column('created_at', sa.DateTime(timezone=True), server_default=sa.text('now()'), nullable=True),
        sa.Column('updated_at', sa.DateTime(timezone=True), nullable=True),
        sa.ForeignKeyConstraint(['user_id'], ['users.id'], ),
        sa.PrimaryKeyConstraint('id'),
        sa.UniqueConstraint('user_id')
    )
    op.create_index(op.f('ix_user_preferences_id'), 'user_preferences', ['id'], unique=False)
    
    # 迁移旧数据
    op.execute("""
        INSERT INTO user_preferences (user_id, preferences)
        SELECT id, '{}' FROM users
    """)


def downgrade():
    op.drop_index(op.f('ix_user_preferences_id'), table_name='user_preferences')
    op.drop_table('user_preferences')
```

## 4. 备份与恢复

### 4.1 自动备份脚本

```bash
#!/bin/bash
# scripts/backup.sh

set -e

# 配置
BACKUP_DIR="/opt/backups/xr-test"
RETENTION_DAYS=30
DATE=$(date +%Y%m%d_%H%M%S)
PROJECT_DIR="/opt/xr-test"

# 创建备份目录
mkdir -p $BACKUP_DIR

# 备份数据库
echo "Backing up database..."
docker-compose -f $PROJECT_DIR/docker-compose.yml exec -T postgres pg_dump \
    -U xruser xr_test_db | gzip > $BACKUP_DIR/db_backup_$DATE.sql.gz

# 备份上传文件
echo "Backing up uploads..."
docker run --rm \
    -v xr-test-platform_uploads:/data \
    -v $BACKUP_DIR:/backup \
    alpine tar czf /backup/uploads_$DATE.tar.gz -C /data .

# 备份配置文件
echo "Backing up configs..."
tar czf $BACKUP_DIR/configs_$DATE.tar.gz \
    -C $PROJECT_DIR \
    .env docker-compose.yml nginx/

# 清理旧备份
echo "Cleaning up old backups..."
find $BACKUP_DIR -name "*.gz" -mtime +$RETENTION_DAYS -delete

echo "Backup completed: $BACKUP_DIR"
```

### 4.2 恢复脚本

```bash
#!/bin/bash
# scripts/restore.sh

set -e

# 配置
BACKUP_DIR="/opt/backups/xr-test"
PROJECT_DIR="/opt/xr-test"

# 列出可用备份
echo "Available backups:"
ls -lt $BACKUP_DIR/*.gz | head -10

# 选择备份文件
read -p "Enter database backup file name: " DB_BACKUP
read -p "Enter uploads backup file name: " UPLOADS_BACKUP

# 停止服务
echo "Stopping services..."
docker-compose -f $PROJECT_DIR/docker-compose.yml down

# 恢复数据库
echo "Restoring database..."
docker-compose -f $PROJECT_DIR/docker-compose.yml up -d postgres
sleep 5
docker-compose -f $PROJECT_DIR/docker-compose.yml exec -T postgres psql \
    -U xruser -c "DROP DATABASE IF EXISTS xr_test_db; CREATE DATABASE xr_test_db;"
gunzip < $BACKUP_DIR/$DB_BACKUP | docker-compose -f $PROJECT_DIR/docker-compose.yml exec -T postgres psql -U xruser xr_test_db

# 恢复上传文件
echo "Restoring uploads..."
docker run --rm \
    -v xr-test-platform_uploads:/data \
    -v $BACKUP_DIR:/backup \
    alpine sh -c "rm -rf /data/* && tar xzf /backup/$UPLOADS_BACKUP -C /data"

# 启动服务
echo "Starting services..."
docker-compose -f $PROJECT_DIR/docker-compose.yml up -d

echo "Restore completed!"
```

## 5. 监控与告警

### 5.1 健康检查端点

```python
# backend/app/routers/health.py
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from redis import Redis
from app.database import get_db

router = APIRouter(prefix="/health", tags=["健康检查"])

@router.get("")
async def health_check():
    return {"status": "healthy", "service": "xr-test-platform"}

@router.get("/db")
async def db_health_check(db: Session = Depends(get_db)):
    try:
        db.execute("SELECT 1")
        return {"status": "healthy", "service": "database"}
    except Exception as e:
        return {"status": "unhealthy", "service": "database", "error": str(e)}

@router.get("/redis")
async def redis_health_check():
    try:
        redis = Redis.from_url("redis://redis:6379/0")
        redis.ping()
        return {"status": "healthy", "service": "redis"}
    except Exception as e:
        return {"status": "unhealthy", "service": "redis", "error": str(e)}
```

## 6. SSL/TLS配置

### 6.1 使用Let's Encrypt

```bash
# 安装certbot
docker run -it --rm \
    -v "$(pwd)/nginx/ssl:/etc/letsencrypt" \
    -v "$(pwd)/nginx/www:/var/www/certbot" \
    certbot/certbot certonly \
    --standalone \
    -d your-domain.com \
    -d www.your-domain.com

# 自动续期脚本 (添加到crontab)
0 12 * * * docker run -it --rm \
    -v "$(pwd)/nginx/ssl:/etc/letsencrypt" \
    -v "$(pwd)/nginx/www:/var/www/certbot" \
    certbot/certbot renew --quiet
```

### 6.2 Nginx SSL配置

```nginx
# nginx/nginx.conf
server {
    listen 80;
    server_name your-domain.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-domain.com;

    ssl_certificate /etc/nginx/ssl/live/your-domain.com/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/live/your-domain.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;

    # 前端静态文件
    location / {
        root /usr/share/nginx/html;
        try_files $uri $uri/ /index.html;
    }

    # API代理
    location /api/ {
        proxy_pass http://backend:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
```

```bash

docker-compose exec postgres psql -U xruser -l
```

## 8. 升级指南

### 8.1 升级后端版本

```bash
# 1. 备份数据
./scripts/backup.sh

# 2. 拉取新代码
git pull origin main

# 3. 重新构建镜像
docker-compose build backend

# 4. 执行数据库迁移
docker-compose run --rm backend alembic upgrade head

# 5. 重启服务
docker-compose up -d backend

# 6. 验证升级
curl http://localhost/api/v1/health
```

### 8.2 升级前端版本

```bash
# 1. 拉取新代码
git pull origin main

# 2. 构建新版本
cd frontend
npm install
npm run build

# 3. 更新Nginx容器
docker-compose up -d --build frontend

# 4. 清除浏览器缓存
# 用户需按Ctrl+F5强制刷新
```

### 8.3 升级数据库版本

```bash
# 1. 备份数据
./scripts/backup.sh

# 2. 修改docker-compose.yml中的PostgreSQL版本
# image: postgres:15-alpine → postgres:16-alpine

# 3. 导出数据
docker-compose exec postgres pg_dump -U xruser xr_test_db > upgrade_backup.sql

# 4. 停止并删除旧容器
docker-compose down

# 5. 删除旧数据卷（谨慎！）
docker volume rm xr-test-platform_postgres_data

# 6. 启动新容器
docker-compose up -d postgres

# 7. 恢复数据
docker-compose exec -T postgres psql -U xruser xr_test_db < upgrade_backup.sql

# 8. 启动其他服务
docker-compose up -d
```
