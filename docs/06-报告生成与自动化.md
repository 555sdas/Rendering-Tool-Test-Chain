```python
from typing import Dict, List, Optional, Any
from sqlalchemy.orm import Session
from datetime import datetime
import json
import os

from app.models.test_report import TestReport, ReportStatus, ReportFormat, ReportTemplate
from app.models.test_session import TestSession
from app.models.performance_analysis import PerformanceAnalysis
from app.models.scene_asset import SceneAsset
from app.services.export_service import ExportService

class ReportGenerationService:
    def __init__(self, db: Session):
        self.db = db
        self.export_service = ExportService()
    
    def create_report(self, project_id: int, template_id: Optional[int], 
                      config: Dict, user_id: int) -> TestReport:
        # 生成报告编号
        report_number = self._generate_report_number()
        
        report = TestReport(
            project_id=project_id,
            template_id=template_id,
            title=config.get("title", "测试报告"),
            report_number=report_number,
            session_ids=config.get("session_ids", []),
            analysis_ids=config.get("analysis_ids", []),
            file_format=ReportFormat(config.get("format", "pdf")),
            generated_by=user_id
        )
        
        self.db.add(report)
        self.db.commit()
        self.db.refresh(report)
        return report
    
    def _generate_report_number(self) -> str:
        import random
        timestamp = datetime.now().strftime("%Y%m%d")
        random_suffix = ''.join([str(random.randint(0, 9)) for _ in range(4)])
        return f"XR-TEST-{timestamp}-{random_suffix}"
    
    def generate_report_content(self, report_id: int) -> Dict:
        report = self.db.query(TestReport).filter(TestReport.id == report_id).first()
        if not report:
            raise ValueError(f"Report {report_id} not found")
        
        report.status = ReportStatus.GENERATING
        self.db.commit()
        
        try:
            # 收集所有相关数据
            sessions = self._collect_session_data(report.session_ids)
            analyses = self._collect_analysis_data(report.analysis_ids)
            
            # 生成报告内容
            content = {
                "report_info": {
                    "title": report.title,
                    "report_number": report.report_number,
                    "generated_at": datetime.now().isoformat(),
                    "version": "1.0"
                },
                "test_summary": self._generate_test_summary(sessions, analyses),
                "test_environment": self._generate_environment_info(sessions),
                "test_scenes": self._generate_scene_info(sessions),
                "performance_results": self._generate_performance_results(analyses),
                "visual_quality_results": self._generate_visual_quality_results(sessions),
                "resource_analysis": self._generate_resource_analysis(sessions),
                "issues_and_recommendations": self._generate_issues_recommendations(analyses),
                "conclusion": self._generate_conclusion(analyses)
            }
            
            report.content = content
            report.status = ReportStatus.COMPLETED
            report.generated_at = datetime.utcnow()
            
        except Exception as e:
            report.status = ReportStatus.FAILED
            report.content = {"error": str(e)}
        
        self.db.commit()
        self.db.refresh(report)
        return report.content
    
    def _collect_session_data(self, session_ids: List[int]) -> List[Dict]:
        sessions = []
        for sid in session_ids:
            session = self.db.query(TestSession).filter(TestSession.id == sid).first()
            if session:
                sessions.append({
                    "id": session.id,
                    "device_info": session.device_info,
                    "scene_info": session.scene_info,
                    "runtime_config": session.runtime_config,
                    "start_time": session.start_time.isoformat() if session.start_time else None,
                    "end_time": session.end_time.isoformat() if session.end_time else None,
                    "summary_stats": session.summary_stats,
                    "status": session.status.value
                })
        return sessions
    
    def _collect_analysis_data(self, analysis_ids: List[int]) -> List[Dict]:
        analyses = []
        for aid in analysis_ids:
            analysis = self.db.query(PerformanceAnalysis).filter(PerformanceAnalysis.id == aid).first()
            if analysis:
                analyses.append({
                    "id": analysis.id,
                    "session_id": analysis.session_id,
                    "avg_fps": analysis.avg_fps,
                    "p95_frame_time": analysis.p95_frame_time,
                    "dropped_frame_rate": analysis.dropped_frame_rate,
                    "long_frame_count": analysis.long_frame_count,
                    "peak_cpu_percent": analysis.peak_cpu_percent,
                    "peak_gpu_percent": analysis.peak_gpu_percent,
                    "peak_memory_mb": analysis.peak_memory_mb,
                    "issues": analysis.issues,
                    "recommendations": analysis.recommendations
                })
        return analyses
    
    def _generate_test_summary(self, sessions: List[Dict], analyses: List[Dict]) -> Dict:
        total_sessions = len(sessions)
        completed_sessions = sum(1 for s in sessions if s["status"] == "completed")
        
        avg_fps_values = [a["avg_fps"] for a in analyses if a["avg_fps"]]
        avg_fps = sum(avg_fps_values) / len(avg_fps_values) if avg_fps_values else None
        
        return {
            "total_test_sessions": total_sessions,
            "completed_sessions": completed_sessions,
            "failed_sessions": total_sessions - completed_sessions,
            "overall_avg_fps": round(avg_fps, 2) if avg_fps else None,
            "test_duration_minutes": self._calculate_total_duration(sessions)
        }
    
    def _generate_environment_info(self, sessions: List[Dict]) -> Dict:
        if not sessions:
            return {}
        
        # 使用第一个会话的设备信息
        device_info = sessions[0].get("device_info", {})
        runtime_config = sessions[0].get("runtime_config", {})
        
        return {
            "device_model": device_info.get("deviceModel", "Unknown"),
            "operating_system": device_info.get("operatingSystem", "Unknown"),
            "processor": device_info.get("processorType", "Unknown"),
            "graphics_device": device_info.get("graphicsDeviceName", "Unknown"),
            "system_memory_mb": device_info.get("systemMemorySize", "Unknown"),
            "graphics_memory_mb": device_info.get("graphicsMemorySize", "Unknown"),
            "unity_version": device_info.get("unityVersion", "Unknown"),
            "render_pipeline": device_info.get("renderPipeline", "Unknown"),
            "target_frame_rate": device_info.get("targetFrameRate", "Unknown"),
            "screen_resolution": f"{device_info.get('screenWidth', 0)}x{device_info.get('screenHeight', 0)}"
        }
    
    def _generate_scene_info(self, sessions: List[Dict]) -> List[Dict]:
        scenes = []
        for session in sessions:
            scene_info = session.get("scene_info", {})
            if scene_info:
                scenes.append({
                    "name": scene_info.get("name", "Unknown"),
                    "description": scene_info.get("description", ""),
                    "complexity": scene_info.get("complexity", "unknown")
                })
        return scenes
    
    def _generate_performance_results(self, analyses: List[Dict]) -> Dict:
        if not analyses:
            return {}
        
        fps_data = [a["avg_fps"] for a in analyses if a["avg_fps"]]
        frame_time_data = [a["p95_frame_time"] for a in analyses if a["p95_frame_time"]]
        
        return {
            "fps_statistics": {
                "average": round(sum(fps_data) / len(fps_data), 2) if fps_data else None,
                "min": round(min(fps_data), 2) if fps_data else None,
                "max": round(max(fps_data), 2) if fps_data else None
            },
            "frame_time_statistics": {
                "p95_average_ms": round(sum(frame_time_data) / len(frame_time_data), 2) if frame_time_data else None,
                "min_p95_ms": round(min(frame_time_data), 2) if frame_time_data else None,
                "max_p95_ms": round(max(frame_time_data), 2) if frame_time_data else None
            },
            "stability": {
                "total_dropped_frames": sum(a.get("dropped_frame_rate", 0) or 0 for a in analyses),
                "total_long_frames": sum(a.get("long_frame_count", 0) or 0 for a in analyses)
            },
            "resource_usage": {
                "peak_cpu_percent": max((a.get("peak_cpu_percent", 0) or 0 for a in analyses), default=0),
                "peak_gpu_percent": max((a.get("peak_gpu_percent", 0) or 0 for a in analyses), default=0),
                "peak_memory_mb": max((a.get("peak_memory_mb", 0) or 0 for a in analyses), default=0)
            }
        }
    
    def _generate_visual_quality_results(self, sessions: List[Dict]) -> Dict:
        # 视觉质量结果（从会话数据中提取）
        return {
            "screenshots_count": sum(len(s.get("screenshots", [])) for s in sessions),
            "visual_defects": [],  # 从视觉检测服务获取
            "image_comparisons": []  # 从图像对比服务获取
        }
    
    def _generate_resource_analysis(self, sessions: List[Dict]) -> Dict:
        # 资源分析结果
        return {
            "scene_complexity": "medium",
            "draw_calls_average": 0,
            "triangle_count": 0,
            "texture_memory_mb": 0
        }
    
    def _generate_issues_recommendations(self, analyses: List[Dict]) -> Dict:
        all_issues = []
        all_recommendations = []
        
        for analysis in analyses:
            issues = analysis.get("issues", []) or []
            recommendations = analysis.get("recommendations", []) or []
            
            all_issues.extend(issues)
            all_recommendations.extend(recommendations)
        
        # 去重
        unique_issues = {json.dumps(i, sort_keys=True): i for i in all_issues}
        unique_recommendations = {json.dumps(r, sort_keys=True): r for r in all_recommendations}
        
        return {
            "issues": list(unique_issues.values()),
            "recommendations": list(unique_recommendations.values()),
            "critical_issues": sum(1 for i in all_issues if i.get("severity") == "critical"),
            "high_issues": sum(1 for i in all_issues if i.get("severity") == "high"),
            "medium_issues": sum(1 for i in all_issues if i.get("severity") == "medium"),
            "low_issues": sum(1 for i in all_issues if i.get("severity") == "low")
        }
    
    def _generate_conclusion(self, analyses: List[Dict]) -> Dict:
        critical_count = sum(
            sum(1 for i in (a.get("issues") or []) if i.get("severity") == "critical")
            for a in analyses
        )
        high_count = sum(
            sum(1 for i in (a.get("issues") or []) if i.get("severity") == "high")
            for a in analyses
        )
        
        if critical_count > 0:
            overall_status = "不通过"
            conclusion = "测试未通过，存在严重问题需要立即修复"
        elif high_count > 0:
            overall_status = "有条件通过"
            conclusion = "测试基本通过，但存在重要问题建议优化"
        else:
            overall_status = "通过"
            conclusion = "测试通过，性能表现符合预期"
        
        return {
            "overall_status": overall_status,
            "conclusion": conclusion,
            "critical_issues": critical_count,
            "high_issues": high_count
        }
    
    def _calculate_total_duration(self, sessions: List[Dict]) -> float:
        total_seconds = 0
        for session in sessions:
            start = session.get("start_time")
            end = session.get("end_time")
            if start and end:
                from datetime import datetime
                start_dt = datetime.fromisoformat(start.replace('Z', '+00:00'))
                end_dt = datetime.fromisoformat(end.replace('Z', '+00:00'))
                total_seconds += (end_dt - start_dt).total_seconds()
        return round(total_seconds / 60, 2)
    
    def export_report(self, report_id: int, format_type: str) -> str:
        report = self.db.query(TestReport).filter(TestReport.id == report_id).first()
        if not report:
            raise ValueError(f"Report {report_id} not found")
        
        if not report.content:
            raise ValueError("Report content not generated yet")
        
        output_dir = "./reports"
        os.makedirs(output_dir, exist_ok=True)
        
        filename = f"{report.report_number}.{format_type}"
        filepath = os.path.join(output_dir, filename)
        
        if format_type == "json":
            with open(filepath, 'w', encoding='utf-8') as f:
                json.dump(report.content, f, ensure_ascii=False, indent=2)
        
        elif format_type == "pdf":
            self._generate_pdf_report(report.content, filepath)
        
        elif format_type == "word":
            self._generate_word_report(report.content, filepath)
        
        report.file_path = filepath
        report.file_size_bytes = os.path.getsize(filepath)
        self.db.commit()
        
        return filepath
```

### 3.2 阈值评估服务 (backend/app/services/threshold_evaluation_service.py)

```python
from typing import Dict, List, Optional
from sqlalchemy.orm import Session

from app.models.threshold_rule import ThresholdRule, ThresholdOperator, RiskLevel
from app.models.performance_analysis import PerformanceAnalysis

class ThresholdEvaluationService:
    def __init__(self, db: Session):
        self.db = db
    
    def evaluate_analysis(self, analysis_id: int, scene_type: Optional[str] = None,
                         device_type: Optional[str] = None) -> Dict:
        analysis = self.db.query(PerformanceAnalysis).filter(
            PerformanceAnalysis.id == analysis_id
        ).first()
        
        if not analysis:
            raise ValueError(f"Analysis {analysis_id} not found")
        
        # 获取适用的阈值规则
        rules = self._get_applicable_rules(scene_type, device_type)
        
        results = []
        overall_risk = RiskLevel.PASS
        total_score = 0
        max_score = 0
        
        for rule in rules:
            metric_value = self._get_metric_value(analysis, rule.metric_type)
            if metric_value is None:
                continue
            
            passed = self._evaluate_rule(rule, metric_value)
            
            if not passed:
                rule_risk = rule.risk_level
                if self._risk_priority(rule_risk) > self._risk_priority(overall_risk):
                    overall_risk = rule_risk
            
            max_score += rule.score_weight
            if passed:
                total_score += rule.score_weight
            
            results.append({
                "rule_id": rule.id,
                "rule_name": rule.name,
                "metric_type": rule.metric_type,
                "metric_value": metric_value,
                "threshold": rule.threshold_value,
                "operator": rule.operator.value,
                "passed": passed,
                "risk_level": rule.risk_level.value if not passed else RiskLevel.PASS.value
            })
        
        score_percentage = (total_score / max_score * 100) if max_score > 0 else 100
        
        return {
            "analysis_id": analysis_id,
            "overall_risk_level": overall_risk.value,
            "score": round(score_percentage, 2),
            "total_rules": len(rules),
            "passed_rules": sum(1 for r in results if r["passed"]),
            "failed_rules": sum(1 for r in results if not r["passed"]),
            "rule_results": results
        }
    
    def _get_applicable_rules(self, scene_type: Optional[str], 
                              device_type: Optional[str]) -> List[ThresholdRule]:
        query = self.db.query(ThresholdRule).filter(ThresholdRule.is_active == 1)
        
        rules = query.all()
        
        # 过滤适用条件
        applicable = []
        for rule in rules:
            scene_types = rule.scene_types or []
            device_types = rule.device_types or []
            
            if scene_type and scene_types and scene_type not in scene_types:
                continue
            if device_type and device_types and device_type not in device_types:
                continue
            
            applicable.append(rule)
        
        return applicable
    
    def _get_metric_value(self, analysis: PerformanceAnalysis, metric_type: str) -> Optional[float]:
        metric_map = {
            "avg_fps": analysis.avg_fps,
            "min_fps": analysis.min_fps,
            "p95_frame_time": analysis.p95_frame_time,
            "p99_frame_time": analysis.p99_frame_time,
            "dropped_frame_rate": analysis.dropped_frame_rate,
            "long_frame_rate": analysis.long_frame_rate,
            "avg_cpu_percent": analysis.peak_cpu_percent,  # 使用峰值作为参考
            "avg_gpu_percent": analysis.peak_gpu_percent,
            "peak_memory_mb": analysis.peak_memory_mb,
            "peak_vram_mb": analysis.peak_vram_mb
        }
        
        return metric_map.get(metric_type)
    
    def _evaluate_rule(self, rule: ThresholdRule, value: float) -> bool:
        operator = rule.operator
        threshold = rule.threshold_value
        
        if operator == ThresholdOperator.GREATER_THAN:
            return value > threshold
        elif operator == ThresholdOperator.LESS_THAN:
            return value < threshold
        elif operator == ThresholdOperator.EQUAL:
            return value == threshold
        elif operator == ThresholdOperator.GREATER_EQUAL:
            return value >= threshold
        elif operator == ThresholdOperator.LESS_EQUAL:
            return value <= threshold
        elif operator == ThresholdOperator.NOT_EQUAL:
            return value != threshold
        
        return False
    
    def _risk_priority(self, risk: RiskLevel) -> int:
        priorities = {
            RiskLevel.PASS: 0,
            RiskLevel.LOW: 1,
            RiskLevel.MEDIUM: 2,

        }
        return priorities.get(risk, 0)
```

```python
    def __init__(self, db: Session):
        self.db = db
        self.task_service = TaskOrchestrationService(db)
    
    def create_automation_script(self, config: Dict, user_id: int) -> Dict:
        script = {
            "name": config.get("name", "自动化测试脚本"),
            "description": config.get("description", ""),
            "version": "1.0",
            "steps": []
        }
        
        # 构建自动化步骤
        steps = config.get("steps", [])
        for i, step in enumerate(steps):
            script["steps"].append({
                "step_id": i + 1,
                "action": step.get("action"),
                "target": step.get("target"),
                "params": step.get("params", {}),
                "retry_count": step.get("retry_count", 0),
                "on_failure": step.get("on_failure", "abort")
            })
        
        return script
    
    def execute_automation_script(self, script: Dict, project_id: int, 
                                  user_id: int) -> Dict:
        execution_log = {
            "start_time": datetime.utcnow().isoformat(),
            "steps": [],
            "status": "running"
        }
        
        for step in script.get("steps", []):
            step_result = self._execute_step(step, project_id, user_id)
            execution_log["steps"].append(step_result)
            
            if not step_result["success"]:
                if step.get("on_failure") == "abort":
                    execution_log["status"] = "failed"
                    execution_log["error"] = step_result.get("error")
                    break
                elif step.get("on_failure") == "continue":
                    continue
                elif step.get("on_failure") == "retry":
                    # 重试逻辑
                    for retry in range(step.get("retry_count", 0)):
                        step_result = self._execute_step(step, project_id, user_id)
                        if step_result["success"]:
                            break
        
        if execution_log["status"] == "running":
            execution_log["status"] = "completed"
        
        execution_log["end_time"] = datetime.utcnow().isoformat()
        return execution_log
    
    def _execute_step(self, step: Dict, project_id: int, user_id: int) -> Dict:
        action = step.get("action")
        params = step.get("params", {})
        
        try:
            if action == "launch_app":
                return self._action_launch_app(params)
            elif action == "switch_scene":
                return self._action_switch_scene(params)
            elif action == "start_collection":
                return self._action_start_collection(project_id, params, user_id)
            elif action == "stop_collection":
                return self._action_stop_collection(params)
            elif action == "run_test":
                return self._action_run_test(project_id, params, user_id)
            elif action == "wait":
                return self._action_wait(params)
            elif action == "capture_screenshot":
                return self._action_capture_screenshot(params)
            elif action == "generate_report":
                return self._action_generate_report(project_id, params, user_id)
            else:
                return {"success": False, "error": f"Unknown action: {action}"}
        except Exception as e:
            return {"success": False, "error": str(e)}
    
    def _action_launch_app(self, params: Dict) -> Dict:
        # 启动应用逻辑
        return {"success": True, "message": f"应用已启动: {params.get('app_name', 'Unknown')}"}
    
    def _action_switch_scene(self, params: Dict) -> Dict:
        scene_name = params.get("scene_name")
        return {"success": True, "message": f"场景已切换: {scene_name}"}

        return {"success": True, "message": "数据采集已开始"}

            "name": params.get("test_name", "自动化测试"),
            "task_type": params.get("task_type", "performance"),
            "config": params.get("config", {}),
            "target_scene_ids": params.get("scene_ids", []),
            "target_device_ids": params.get("device_ids", [])
        }
        
        task = self.task_service.create_task(task_data, user_id)
        result = self.task_service.execute_task(task.id)
        
        return {"success": True, "task_id": task.id, "message": "测试任务已启动"}
    
    def _action_wait(self, params: Dict) -> Dict:
        import time
        duration = params.get("duration_seconds", 1)
        time.sleep(duration)
        return {"success": True, "message": f"等待 {duration} 秒完成"}
    
    def _action_capture_screenshot(self, params: Dict) -> Dict:
        return {"success": True, "message": "截图已捕获"}
    
    def _action_generate_report(self, project_id: int, params: Dict, user_id: int) -> Dict:
        return {"success": True, "message": "报告生成任务已启动"}
```
