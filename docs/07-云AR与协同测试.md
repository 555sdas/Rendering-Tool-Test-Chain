# 阶段七：云AR与协同测试

## 1. 目标

实现云AR端云协同与多人空间协同测试能力，包括：
- 云AR端云协同基础测试（单用户/多用户）
- 云端渲染与本地显示稳定性测试
- 空间交互响应测试（手势/语音/物体/控制器）
- 多用户空间协同与集群场景测试
- 典型云AR业务场景与可控干扰测试
- 报告生成与技术培训支持

import enum

class CloudARSessionStatus(str, enum.Enum):
    INITIALIZING = "initializing"
    CONNECTING = "connecting"
    CONNECTED = "connected"
    TESTING = "testing"
    DISCONNECTED = "disconnected"
    COMPLETED = "completed"
    ERROR = "error"

class CloudARSession(Base):
    __tablename__ = "cloud_ar_sessions"
    
    id = Column(Integer, primary_key=True, index=True)
    session_id = Column(String(100), unique=True, index=True, nullable=False)
    project_id = Column(Integer, ForeignKey("test_projects.id"), nullable=False)
    
    # 测试配置
    test_type = Column(String(50), nullable=False)  # single_user, multi_user, stress
    protocol = Column(String(50), default="webrtc")  # webrtc, cloudxr, custom
    rendering_mode = Column(String(50), default="cloud")  # cloud, edge, local_composite
    
    # 参与用户
    user_count = Column(Integer, default=1)
    user_configs = Column(JSON, default=[])  # [{"user_id": "u1", "device_type": "...", "role": "..."}]
    
    # 网络配置
    network_config = Column(JSON, default={})  # {"bandwidth_mbps": 50, "latency_ms": 20, "packet_loss": 0}
    
    # 时间
    start_time = Column(DateTime(timezone=True))
    end_time = Column(DateTime(timezone=True))
    
    # 状态
    status = Column(Enum(CloudARSessionStatus), default=CloudARSessionStatus.INITIALIZING)
    error_message = Column(String(2000))
    
    # 汇总结果
    summary = Column(JSON)
    
    created_by = Column(Integer, ForeignKey("users.id"))
    created_at = Column(DateTime(timezone=True), server_default=func.now())

### 4.1 云AR测试核心服务 (backend/app/services/cloud_ar_test_service.py)

```python
from typing import Dict, List, Optional
from sqlalchemy.orm import Session
from datetime import datetime
import uuid

from app.models.cloud_ar_session import CloudARSession, CloudARSessionStatus
from app.models.cloud_ar_sample import CloudARSample
from app.models.spatial_sync_sample import SpatialSyncSample
from app.models.interaction_event import InteractionEvent

class CloudARTestService:
    def __init__(self, db: Session):
        self.db = db
    
    def create_session(self, project_id: int, config: Dict, user_id: int) -> CloudARSession:
        session = CloudARSession(
            session_id=str(uuid.uuid4()),
            project_id=project_id,
            test_type=config.get("test_type", "single_user"),
            protocol=config.get("protocol", "webrtc"),
            rendering_mode=config.get("rendering_mode", "cloud"),
            user_count=config.get("user_count", 1),
            user_configs=config.get("user_configs", []),
            network_config=config.get("network_config", {}),
            status=CloudARSessionStatus.INITIALIZING,
            created_by=user_id
        )
        self.db.add(session)
        self.db.commit()
        self.db.refresh(session)
        return session
    
    def start_test(self, session_id: int):
        session = self.db.query(CloudARSession).filter(CloudARSession.id == session_id).first()
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        session.status = CloudARSessionStatus.TESTING
        session.start_time = datetime.utcnow()
        self.db.commit()
    
    def record_cloud_ar_sample(self, session_id: int, sample_data: Dict):
        sample = CloudARSample(
            session_id=session_id,
            user_id=sample_data["user_id"],
            timestamp=datetime.fromisoformat(sample_data["timestamp"]),
            source=sample_data.get("source", "client"),
            local_fps=sample_data.get("local_fps"),
            remote_fps=sample_data.get("remote_fps"),
            end_to_end_latency_ms=sample_data.get("end_to_end_latency_ms"),
            encoding_latency_ms=sample_data.get("encoding_latency_ms"),
            decoding_latency_ms=sample_data.get("decoding_latency_ms"),
            network_latency_ms=sample_data.get("network_latency_ms"),
            bandwidth_mbps=sample_data.get("bandwidth_mbps"),
            packet_loss_rate=sample_data.get("packet_loss_rate"),
            jitter_ms=sample_data.get("jitter_ms"),
            rtt_ms=sample_data.get("rtt_ms"),
            resolution=sample_data.get("resolution"),
            bitrate_kbps=sample_data.get("bitrate_kbps"),
            frame_drop_count=sample_data.get("frame_drop_count", 0),
            freeze_detected=sample_data.get("freeze_detected", False),
            black_screen_detected=sample_data.get("black_screen_detected", False),
            quality_degradation=sample_data.get("quality_degradation", False),
            raw_metrics=sample_data.get("raw_metrics", {})
        )
        self.db.add(sample)
        self.db.commit()
    
    def record_spatial_sync(self, session_id: int, sync_data: Dict):
        sync = SpatialSyncSample(
            session_id=session_id,
            timestamp=datetime.fromisoformat(sync_data["timestamp"]),
            user_id=sync_data["user_id"],
            position=sync_data.get("position"),
            rotation=sync_data.get("rotation"),
            sync_delay_ms=sync_data.get("sync_delay_ms"),
            position_drift=sync_data.get("position_drift"),
            rotation_drift=sync_data.get("rotation_drift"),
            virtual_objects=sync_data.get("virtual_objects", []),
            conflict_detected=sync_data.get("conflict_detected", False),
            conflict_type=sync_data.get("conflict_type"),
            conflict_details=sync_data.get("conflict_details")
        )
        self.db.add(sync)
        self.db.commit()
    
    def record_interaction(self, session_id: int, event_data: Dict):
        event = InteractionEvent(
            session_id=session_id,
            user_id=event_data["user_id"],
            timestamp=datetime.fromisoformat(event_data["timestamp"]),
            interaction_type=event_data["interaction_type"],
            action=event_data["action"],
            triggered=event_data.get("triggered", False),
            false_trigger=event_data.get("false_trigger", False),
            missed_trigger=event_data.get("missed_trigger", False),
            response_time_ms=event_data.get("response_time_ms"),
            input_data=event_data.get("input_data"),
            result_data=event_data.get("result_data"),
            confidence=event_data.get("confidence"),
            error_type=event_data.get("error_type"),
            error_message=event_data.get("error_message")
        )
        self.db.add(event)
        self.db.commit()
    
    def complete_session(self, session_id: int):
        session = self.db.query(CloudARSession).filter(CloudARSession.id == session_id).first()
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        session.status = CloudARSessionStatus.COMPLETED
        session.end_time = datetime.utcnow()
        
        # 生成汇总
        session.summary = self._generate_summary(session_id)
        self.db.commit()
    
    def _generate_summary(self, session_id: int) -> Dict:
        # 端云性能汇总
        samples = self.db.query(CloudARSample).filter(
            CloudARSample.session_id == session_id
        ).all()
        
        latency_values = [s.end_to_end_latency_ms for s in samples if s.end_to_end_latency_ms]
        fps_values = [s.local_fps for s in samples if s.local_fps]
        
        # 空间同步汇总
        sync_samples = self.db.query(SpatialSyncSample).filter(
            SpatialSyncSample.session_id == session_id
        ).all()
        
        sync_delays = [s.sync_delay_ms for s in sync_samples if s.sync_delay_ms]
        conflicts = sum(1 for s in sync_samples if s.conflict_detected)
        
        # 交互汇总
        events = self.db.query(InteractionEvent).filter(
            InteractionEvent.session_id == session_id
        ).all()
        
        total_interactions = len(events)
        successful = sum(1 for e in events if e.triggered and not e.false_trigger)
        false_triggers = sum(1 for e in events if e.false_trigger)
        missed_triggers = sum(1 for e in events if e.missed_trigger)
        response_times = [e.response_time_ms for e in events if e.response_time_ms]
        
        return {
            "performance": {
                "avg_latency_ms": round(sum(latency_values) / len(latency_values), 2) if latency_values else None,
                "max_latency_ms": round(max(latency_values), 2) if latency_values else None,
                "avg_fps": round(sum(fps_values) / len(fps_values), 2) if fps_values else None,
                "min_fps": round(min(fps_values), 2) if fps_values else None,
                "freeze_count": sum(1 for s in samples if s.freeze_detected),
                "black_screen_count": sum(1 for s in samples if s.black_screen_detected)
            },
            "spatial_sync": {
                "avg_sync_delay_ms": round(sum(sync_delays) / len(sync_delays), 2) if sync_delays else None,
                "max_sync_delay_ms": round(max(sync_delays), 2) if sync_delays else None,
                "conflict_count": conflicts,
                "position_drift_avg": round(
                    sum(s.position_drift for s in sync_samples if s.position_drift) / 
                    len([s for s in sync_samples if s.position_drift]), 4
                ) if any(s.position_drift for s in sync_samples) else None
            },
            "interaction": {
                "total_events": total_interactions,
                "success_rate": round(successful / total_interactions * 100, 2) if total_interactions > 0 else 0,
                "false_trigger_rate": round(false_triggers / total_interactions * 100, 2) if total_interactions > 0 else 0,
                "missed_trigger_rate": round(missed_triggers / total_interactions * 100, 2) if total_interactions > 0 else 0,
                "avg_response_time_ms": round(sum(response_times) / len(response_times), 2) if response_times else None
            }
        }

# backend/app/services/interference_engine.py

from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum
import random

class InterferenceType(str, Enum):
    BANDWIDTH_LIMIT = "bandwidth_limit"
    LATENCY_INCREASE = "latency_increase"
    PACKET_LOSS = "packet_loss"
    JITTER = "jitter"
    DISCONNECT = "disconnect"
    NOISE = "noise"
    LIGHTING_CHANGE = "lighting_change"
    OCCLUSION = "occlusion"

@dataclass
class InterferenceConfig:
    type: InterferenceType
    duration_seconds: float
    intensity: float  # 0.0 - 1.0
    parameters: Dict

class InterferenceEngine:
    def __init__(self):
        self.active_interferences: List[InterferenceConfig] = []
    
    def apply_interference(self, config: InterferenceConfig) -> Dict:
        result = {
            "type": config.type.value,
            "duration": config.duration_seconds,
            "intensity": config.intensity,
            "applied_at": datetime.utcnow().isoformat(),
            "effects": {}
        }
        
        if config.type == InterferenceType.BANDWIDTH_LIMIT:
            result["effects"] = self._apply_bandwidth_limit(config)
        elif config.type == InterferenceType.LATENCY_INCREASE:
            result["effects"] = self._apply_latency_increase(config)
        elif config.type == InterferenceType.PACKET_LOSS:
            result["effects"] = self._apply_packet_loss(config)
        elif config.type == InterferenceType.JITTER:
            result["effects"] = self._apply_jitter(config)
        elif config.type == InterferenceType.DISCONNECT:
            result["effects"] = self._apply_disconnect(config)
        elif config.type == InterferenceType.NOISE:
            result["effects"] = self._apply_noise(config)
        elif config.type == InterferenceType.LIGHTING_CHANGE:
            result["effects"] = self._apply_lighting_change(config)
        elif config.type == InterferenceType.OCCLUSION:
            result["effects"] = self._apply_occlusion(config)
        
        self.active_interferences.append(config)
        return result
    
    def _apply_bandwidth_limit(self, config: InterferenceConfig) -> Dict:
        limit_mbps = config.parameters.get("limit_mbps", 10)
        return {
            "bandwidth_limit_mbps": limit_mbps,
            "expected_quality_degradation": config.intensity * 100,
            "description": f"带宽限制为 {limit_mbps} Mbps"
        }
    
    def _apply_latency_increase(self, config: InterferenceConfig) -> Dict:
        additional_ms = config.parameters.get("additional_ms", 100)
        return {
            "additional_latency_ms": additional_ms,
            "expected_e2e_latency_ms": 20 + additional_ms,
            "description": f"增加 {additional_ms}ms 延迟"
        }
    
    def _apply_packet_loss(self, config: InterferenceConfig) -> Dict:
        loss_rate = config.parameters.get("loss_rate", 0.05)
        return {
            "packet_loss_rate": loss_rate,
            "expected_frame_corruption": loss_rate * 2,
            "description": f"丢包率设置为 {loss_rate * 100}%"
        }
    
    def _apply_jitter(self, config: InterferenceConfig) -> Dict:
        jitter_ms = config.parameters.get("jitter_ms", 50)
        return {
            "jitter_ms": jitter_ms,
            "expected_stability_impact": "high" if jitter_ms > 30 else "medium",
            "description": f"抖动设置为 {jitter_ms}ms"
        }
    
    def _apply_disconnect(self, config: InterferenceConfig) -> Dict:
        duration_ms = config.parameters.get("duration_ms", 5000)
        return {
            "disconnect_duration_ms": duration_ms,
            "expected_reconnection_time_ms": duration_ms * 2,
            "description": f"断开连接 {duration_ms}ms"
        }
    
    def _apply_noise(self, config: InterferenceConfig) -> Dict:
        noise_level = config.parameters.get("noise_level", 0.3)
        return {
            "noise_level": noise_level,
            "expected_tracking_accuracy_drop": noise_level * 50,
            "description": f"添加 {noise_level * 100}% 噪声"
        }
    
    def _apply_lighting_change(self, config: InterferenceConfig) -> Dict:
        brightness_change = config.parameters.get("brightness_change", -50)
        return {
            "brightness_change_percent": brightness_change,
            "expected_tracking_impact": "high" if abs(brightness_change) > 40 else "medium",
            "description": f"亮度变化 {brightness_change}%"
        }
    
    def _apply_occlusion(self, config: InterferenceConfig) -> Dict:
        occlusion_ratio = config.parameters.get("occlusion_ratio", 0.5)
        return {
            "occlusion_ratio": occlusion_ratio,
            "expected_tracking_loss": occlusion_ratio > 0.3,
            "description": f"遮挡比例 {occlusion_ratio * 100}%"
        }
    
    def generate_interference_scenario(self, scenario_name: str) -> List[InterferenceConfig]:
        scenarios = {
            "weak_network": [
                InterferenceConfig(
                    type=InterferenceType.BANDWIDTH_LIMIT,
                    duration_seconds=60,
                    intensity=0.5,
                    parameters={"limit_mbps": 5}
                ),
                InterferenceConfig(
                    type=InterferenceType.PACKET_LOSS,
                    duration_seconds=30,
                    intensity=0.3,
                    parameters={"loss_rate": 0.03}
                )
            ],
            "network_recovery": [
                InterferenceConfig(
                    type=InterferenceType.DISCONNECT,
                    duration_seconds=10,
                    intensity=0.8,
                    parameters={"duration_ms": 3000}
                ),
                InterferenceConfig(
                    type=InterferenceType.LATENCY_INCREASE,
                    duration_seconds=30,
                    intensity=0.6,
                    parameters={"additional_ms": 200}
                )
            ],
            "environmental_challenge": [
                InterferenceConfig(
                    type=InterferenceType.LIGHTING_CHANGE,
                    duration_seconds=45,
                    intensity=0.7,
                    parameters={"brightness_change": -60}
                ),
                InterferenceConfig(
                    type=InterferenceType.OCCLUSION,
                    duration_seconds=20,
                    intensity=0.5,
                    parameters={"occlusion_ratio": 0.4}
                )
            ]
        }
        return scenarios.get(scenario_name, [])

            "report_type": "cloud_ar_test",
            "session_info": {
                "session_id": session.session_id,
                "test_type": session.test_type,
                "protocol": session.protocol,
                "rendering_mode": session.rendering_mode,
                "user_count": session.user_count,
                "start_time": session.start_time.isoformat() if session.start_time else None,
                "end_time": session.end_time.isoformat() if session.end_time else None
            },
            "test_environment": {
                "network_config": session.network_config,
                "user_configs": session.user_configs
            },
            "performance_analysis": self._analyze_performance(session_id),
            "spatial_sync_analysis": self._analyze_spatial_sync(session_id),
            "interaction_analysis": self._analyze_interactions(session_id),
            "stability_assessment": self._assess_stability(session_id),
            "issues_and_recommendations": self._generate_recommendations(session_id)
        }
    
    def _analyze_performance(self, session_id: int) -> Dict:
        samples = self.db.query(CloudARSample).filter(
            CloudARSample.session_id == session_id
        ).all()
        
        if not samples:
            return {}
        
        latencies = [s.end_to_end_latency_ms for s in samples if s.end_to_end_latency_ms]
        fps_values = [s.local_fps for s in samples if s.local_fps]
        
        return {
            "latency": {
                "avg_ms": round(sum(latencies) / len(latencies), 2) if latencies else None,
                "min_ms": round(min(latencies), 2) if latencies else None,
                "max_ms": round(max(latencies), 2) if latencies else None,
                "p95_ms": self._percentile(latencies, 95) if latencies else None
            },
            "frame_rate": {
                "avg_fps": round(sum(fps_values) / len(fps_values), 2) if fps_values else None,
                "min_fps": round(min(fps_values), 2) if fps_values else None,
                "stability": self._calculate_stability(fps_values) if fps_values else None
            },
            "quality_events": {
                "freeze_count": sum(1 for s in samples if s.freeze_detected),
                "black_screen_count": sum(1 for s in samples if s.black_screen_detected),
                "quality_degradation_count": sum(1 for s in samples if s.quality_degradation)
            }
        }
    
    def _analyze_spatial_sync(self, session_id: int) -> Dict:
        samples = self.db.query(SpatialSyncSample).filter(
            SpatialSyncSample.session_id == session_id
        ).all()
        
        if not samples:
            return {}
        
        delays = [s.sync_delay_ms for s in samples if s.sync_delay_ms]
        drifts = [s.position_drift for s in samples if s.position_drift]
        conflicts = [s for s in samples if s.conflict_detected]
        
        return {
            "sync_delay": {
                "avg_ms": round(sum(delays) / len(delays), 2) if delays else None,
                "max_ms": round(max(delays), 2) if delays else None
            },
            "position_drift": {
                "avg_m": round(sum(drifts) / len(drifts), 4) if drifts else None,
                "max_m": round(max(drifts), 4) if drifts else None
            },
            "conflicts": {
                "count": len(conflicts),
                "types": list(set(c.conflict_type for c in conflicts if c.conflict_type))
            }
        }
    
    def _analyze_interactions(self, session_id: int) -> Dict:
        events = self.db.query(InteractionEvent).filter(
            InteractionEvent.session_id == session_id
        ).all()
        
        if not events:
            return {}
        
        by_type = {}
        for event in events:
            itype = event.interaction_type
            if itype not in by_type:
                by_type[itype] = {"total": 0, "success": 0, "false": 0, "missed": 0}
            by_type[itype]["total"] += 1
            if event.triggered and not event.false_trigger:
                by_type[itype]["success"] += 1
            if event.false_trigger:
                by_type[itype]["false"] += 1
            if event.missed_trigger:
                by_type[itype]["missed"] += 1
        
        response_times = [e.response_time_ms for e in events if e.response_time_ms]
        
        return {
            "by_type": {
                k: {
                    "total": v["total"],
                    "success_rate": round(v["success"] / v["total"] * 100, 2) if v["total"] > 0 else 0,
                    "false_trigger_rate": round(v["false"] / v["total"] * 100, 2) if v["total"] > 0 else 0,
                    "missed_trigger_rate": round(v["missed"] / v["total"] * 100, 2) if v["total"] > 0 else 0
                }
                for k, v in by_type.items()
            },
            "response_time": {
                "avg_ms": round(sum(response_times) / len(response_times), 2) if response_times else None,
                "p95_ms": self._percentile(response_times, 95) if response_times else None
            }
        }
    
    def _assess_stability(self, session_id: int) -> Dict:
        samples = self.db.query(CloudARSample).filter(
            CloudARSample.session_id == session_id
        ).all()
        
        total_samples = len(samples)
        if total_samples == 0:
            return {}
        
        abnormal_samples = sum(1 for s in samples if s.freeze_detected or 
                              s.black_screen_detected or s.quality_degradation)
        
        stability_rate = (1 - abnormal_samples / total_samples) * 100
        
        return {
            "stability_rate": round(stability_rate, 2),
            "total_samples": total_samples,
            "abnormal_samples": abnormal_samples,
            "assessment": "稳定" if stability_rate > 95 else "基本稳定" if stability_rate > 85 else "不稳定"
        }
    
    def _generate_recommendations(self, session_id: int) -> List[Dict]:
        recommendations = []
        
        # 基于性能分析生成建议
        perf = self._analyze_performance(session_id)
        if perf.get("latency", {}).get("avg_ms", 0) > 100:
            recommendations.append({
                "category": "latency",
                "priority": "high",
                "issue": "端到端延迟过高",
                "suggestion": "优化编码参数或选择更近的云端节点",
                "details": "建议采用硬件编码、降低分辨率或启用低延迟模式"
            })
        
        # 基于空间同步生成建议
        sync = self._analyze_spatial_sync(session_id)
        if sync.get("sync_delay", {}).get("avg_ms", 0) > 50:
            recommendations.append({
                "category": "spatial_sync",
                "priority": "high",
                "issue": "空间同步延迟过高",
                "suggestion": "优化网络同步策略或采用预测补偿",
                "details": "建议启用客户端预测、服务器回退或插值平滑"
            })
        
        # 基于交互生成建议
        interaction = self._analyze_interactions(session_id)
        for itype, stats in interaction.get("by_type", {}).items():
            if stats.get("success_rate", 100) < 90:
                recommendations.append({
                    "category": "interaction",
                    "priority": "medium",
                    "issue": f"{itype}交互成功率偏低",
                    "suggestion": "优化识别算法或调整交互阈值",
                    "details": f"当前成功率 {stats['success_rate']}%，建议优化"
                })
        
        return recommendations
    
    @staticmethod
    def _percentile(values: List[float], p: float) -> float:
        sorted_values = sorted(values)
        k = (len(sorted_values) - 1) * p / 100
        f = int(k)
        c = f + 1 if f + 1 < len(sorted_values) else f
        if f == c:
            return sorted_values[f]
        return sorted_values[f] * (c - k) + sorted_values[c] * (k - f)
    
    @staticmethod
    def _calculate_stability(values: List[float]) -> float:
        if len(values) < 2:
            return 100.0
        avg = sum(values) / len(values)
        variance = sum((x - avg) ** 2 for x in values) / len(values)
        std_dev = variance ** 0.5
        stability = max(0, 100 - std_dev * 2)
        return round(stability, 2)
```

## 7. 验收标准

| 验收项 | 标准 |
|--------|------|
| 单用户端云交互 | 支持单用户端云交互测试，记录时间戳/帧率/响应时间 |
| 多人协同 | 支持不少于2用户的多人空间协同测试 |
| 云端渲染稳定性 | 记录帧率/帧时间/串流分辨率/编码解码状态/异常帧 |

| 交互指标 | 记录触发成功率/漏触发/误触发/响应时间 |
| 多用户同步 | 记录用户位置/姿态同步/虚拟对象状态同步/同步延迟 |
| 冲突检测 | 检测并记录空间冲突和状态冲突 |
| 业务场景 | 提供不少于2类云AR测试任务 |
| 干扰测试 | 支持带宽波动/弱网/丢包/无网恢复/噪声/光照变化/遮挡 |
| 报告生成 | 生成包含测试对象/环境/用户数量/网络条件/交互结果/空间协同结果的报告 |
| 数据管理 | 支持不少于20条云AR测试会话记录的管理与查询 |
