### 3.1 安全中间件 (backend/app/core/security_middleware.py)

```python
from fastapi import Request, HTTPException, status
from fastapi.responses import JSONResponse
from starlette.middleware.base import BaseHTTPMiddleware
from typing import Optional
import re
import time
from collections import defaultdict

class SecurityMiddleware(BaseHTTPMiddleware):
    def __init__(self, app):
        super().__init__(app)
        # 速率限制: IP -> {endpoint -> [timestamps]}
        self.rate_limit_store = defaultdict(lambda: defaultdict(list))
        self.rate_limit = 100  # 每分钟请求数
        
        # SQL注入特征
        self.sql_injection_patterns = [
            r"(\b(SELECT|INSERT|UPDATE|DELETE|DROP|CREATE|ALTER|EXEC|UNION)\b)",
            r"(\b(OR|AND)\b\s+\d+\s*=\s*\d+)",
            r"(--|#|/\*|\*/)",
            r"(\bWAITFOR\b\s+\bDELAY\b)",
            r"(\bBENCHMARK\b\s*\()")
        ]
        
        # XSS特征
        self.xss_patterns = [
            r"(<script[^>]*>)",
            r"(javascript:)",
            r"(on\w+\s*=)",
            r"(<iframe[^>]*>)",
            r"(<object[^>]*>)")
        ]
    
    async def dispatch(self, request: Request, call_next):
        # 1. 速率限制检查
        if not self._check_rate_limit(request):
            return JSONResponse(
                status_code=status.HTTP_429_TOO_MANY_REQUESTS,
                content={"detail": "请求过于频繁，请稍后重试"}
            )
        
        # 2. 输入验证
        if not self._validate_input(request):
            return JSONResponse(
                status_code=status.HTTP_400_BAD_REQUEST,
                content={"detail": "请求包含非法字符"}
            )
        
        # 3. 安全响应头
        response = await call_next(request)
        response.headers["X-Content-Type-Options"] = "nosniff"
        response.headers["X-Frame-Options"] = "DENY"
        response.headers["X-XSS-Protection"] = "1; mode=block"
        response.headers["Strict-Transport-Security"] = "max-age=31536000; includeSubDomains"
        response.headers["Content-Security-Policy"] = "default-src 'self'; script-src 'self' 'unsafe-inline'"
        
        return response
    
    def _check_rate_limit(self, request: Request) -> bool:
        client_ip = request.client.host
        endpoint = request.url.path
        now = time.time()
        
        # 清理过期记录
        self.rate_limit_store[client_ip][endpoint] = [
            ts for ts in self.rate_limit_store[client_ip][endpoint]
            if now - ts < 60
        ]
        
        if len(self.rate_limit_store[client_ip][endpoint]) >= self.rate_limit:
            return False
        
        self.rate_limit_store[client_ip][endpoint].append(now)
        return True
    
    def _validate_input(self, request: Request) -> bool:
        # 检查URL参数
        query_string = str(request.query_params)
        if self._contains_sql_injection(query_string) or self._contains_xss(query_string):
            return False
        
        # 检查路径
        path = request.url.path
        if self._contains_sql_injection(path) or self._contains_xss(path):
            return False
        
        return True
    
    def _contains_sql_injection(self, text: str) -> bool:
        text_upper = text.upper()
        for pattern in self.sql_injection_patterns:
            if re.search(pattern, text_upper):
                return True
        return False
    
    def _contains_xss(self, text: str) -> bool:
        for pattern in self.xss_patterns:
            if re.search(pattern, text, re.IGNORECASE):
                return True
        return False
```

### 3.2 审计日志服务 (backend/app/services/audit_service.py)

```python
from sqlalchemy.orm import Session
from typing import Optional, Dict, Any
from datetime import datetime
import json
import hashlib

from app.models.audit_log import AuditLog, AuditAction

class AuditService:
    def __init__(self, db: Session):
        self.db = db
    
    async def log(self, action: str, user_id: Optional[int] = None, 
                  username: Optional[str] = None, resource_type: Optional[str] = None,
                  resource_id: Optional[str] = None, details: Optional[Dict] = None,
                  ip_address: Optional[str] = None, user_agent: Optional[str] = None):
        
        log_entry = AuditLog(
            user_id=user_id,
            username=username or "anonymous",
            action=action,
            resource_type=resource_type,
            resource_id=resource_id,
            details=details or {},
            ip_address=ip_address,
            user_agent=user_agent,
            created_at=datetime.utcnow()
        )
        
        self.db.add(log_entry)
        self.db.commit()
        
        # 生成日志完整性校验
        self._generate_log_integrity(log_entry)
    
    def _generate_log_integrity(self, log_entry: AuditLog):
        # 使用前一跳日志的哈希构建链式结构
        prev_log = self.db.query(AuditLog).filter(
            AuditLog.id < log_entry.id
        ).order_by(AuditLog.id.desc()).first()
        
        data = {
            "id": log_entry.id,
            "action": log_entry.action,
            "user_id": log_entry.user_id,
            "timestamp": log_entry.created_at.isoformat(),
            "prev_hash": prev_log.integrity_hash if prev_log else "0"
        }
        
        hash_str = json.dumps(data, sort_keys=True)
        log_entry.integrity_hash = hashlib.sha256(hash_str.encode()).hexdigest()
        self.db.commit()
    
    def verify_log_integrity(self, start_id: Optional[int] = None, 
                            end_id: Optional[int] = None) -> Dict:
        query = self.db.query(AuditLog)
        
        if start_id:
            query = query.filter(AuditLog.id >= start_id)
        if end_id:
            query = query.filter(AuditLog.id <= end_id)
        
        logs = query.order_by(AuditLog.id).all()
        
        violations = []
        for i, log in enumerate(logs):
            if i == 0:
                continue
            
            prev_log = logs[i - 1]
            expected_hash = self._calculate_expected_hash(log, prev_log.integrity_hash)
            
            if log.integrity_hash != expected_hash:
                violations.append({
                    "log_id": log.id,
                    "expected_hash": expected_hash,
                    "actual_hash": log.integrity_hash
                })
        
        return {
            "total_logs": len(logs),
            "violations": violations,
            "is_valid": len(violations) == 0
        }
    
    def _calculate_expected_hash(self, log: AuditLog, prev_hash: str) -> str:
        data = {
            "id": log.id,
            "action": log.action,
            "user_id": log.user_id,
            "timestamp": log.created_at.isoformat(),
            "prev_hash": prev_hash
        }
        return hashlib.sha256(json.dumps(data, sort_keys=True).encode()).hexdigest()
    
    def get_audit_summary(self, start_date: Optional[datetime] = None,
                         end_date: Optional[datetime] = None) -> Dict:
        query = self.db.query(AuditLog)
        
        if start_date:
            query = query.filter(AuditLog.created_at >= start_date)
        if end_date:
            query = query.filter(AuditLog.created_at <= end_date)
        
        logs = query.all()
        
        action_counts = {}
        user_activity = {}
        
        for log in logs:
            action_counts[log.action] = action_counts.get(log.action, 0) + 1
            if log.user_id:
                user_activity[log.user_id] = user_activity.get(log.user_id, 0) + 1
        
        return {
            "total_logs": len(logs),
            "action_distribution": action_counts,
            "user_activity": user_activity,
            "suspicious_activities": self._detect_suspicious_activity(logs)
        }
    
    def _detect_suspicious_activity(self, logs: list) -> list:
        suspicious = []
        
        # 检测多次登录失败
        failed_logins = {}
        for log in logs:
            if log.action == "login_failed":
                key = log.username or log.ip_address
                failed_logins[key] = failed_logins.get(key, 0) + 1
        
        for key, count in failed_logins.items():
            if count >= 5:
                suspicious.append({
                    "type": "multiple_failed_logins",
                    "target": key,
                    "count": count,
                    "severity": "high" if count >= 10 else "medium"
                })
        
        # 检测异常时间访问
        for log in logs:
            hour = log.created_at.hour
            if hour < 6 or hour > 23:
                suspicious.append({
                    "type": "off_hours_access",
                    "user": log.username,
                    "time": log.created_at.isoformat(),
                    "action": log.action,
                    "severity": "low"
                })
        
        return suspicious
```

### 3.3 数据加密服务 (backend/app/services/encryption_service.py)

```python
from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import base64
import os
from typing import Optional

class EncryptionService:
    def __init__(self, master_key: Optional[str] = None):
        self.master_key = master_key or os.environ.get("ENCRYPTION_MASTER_KEY")
        if not self.master_key:
            raise ValueError("Encryption master key is required")
        
        self._init_cipher()
    
    def _init_cipher(self):
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=b"xr_test_platform_salt",
            iterations=100000,
        )
        key = base64.urlsafe_b64encode(kdf.derive(self.master_key.encode()))
        self.cipher = Fernet(key)
    
    def encrypt(self, data: str) -> str:
        return self.cipher.encrypt(data.encode()).decode()
    
    def decrypt(self, encrypted_data: str) -> str:
        return self.cipher.decrypt(encrypted_data.encode()).decode()
    
    def encrypt_file(self, file_path: str, output_path: Optional[str] = None) -> str:
        with open(file_path, 'rb') as f:
            data = f.read()
        
        encrypted = self.cipher.encrypt(data)
        
        output = output_path or f"{file_path}.encrypted"
        with open(output, 'wb') as f:
            f.write(encrypted)
        
        return output
    
    def decrypt_file(self, file_path: str, output_path: Optional[str] = None) -> str:
        with open(file_path, 'rb') as f:
            encrypted = f.read()
        
        decrypted = self.cipher.decrypt(encrypted)
        
        output = output_path or file_path.replace('.encrypted', '')
        with open(output, 'wb') as f:
            f.write(decrypted)
        
        return output
    
    def hash_sensitive_data(self, data: str) -> str:
        import hashlib
        return hashlib.sha256(f"{data}{self.master_key}".encode()).hexdigest()
```

### 3.4 文件上传安全 (backend/app/core/file_security.py)

```python
from fastapi import UploadFile, HTTPException, status
from typing import List, Optional
import magic
import os

class FileSecurityValidator:
    # 允许的文件类型
    ALLOWED_IMAGE_TYPES = {
        'image/png': '.png',
        'image/jpeg': '.jpg',
        'image/jpg': '.jpg',
        'image/gif': '.gif',
        'image/webp': '.webp'
    }
    
    ALLOWED_DOCUMENT_TYPES = {
        'application/pdf': '.pdf',
        'application/json': '.json',
        'text/csv': '.csv',
        'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet': '.xlsx'
    }
    
    ALLOWED_ARCHIVE_TYPES = {
        'application/zip': '.zip',
        'application/x-zip-compressed': '.zip'
    }
    
    # 最大文件大小 (100MB)
    MAX_FILE_SIZE = 100 * 1024 * 1024
    
    @classmethod
    def validate_upload(cls, file: UploadFile, allowed_types: Optional[dict] = None) -> dict:
        allowed = allowed_types or {**cls.ALLOWED_IMAGE_TYPES, **cls.ALLOWED_DOCUMENT_TYPES}
        
        # 检查文件名
        if not file.filename:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="文件名不能为空"
            )
        
        # 检查文件大小
        file.file.seek(0, 2)  # 移动到文件末尾
        file_size = file.file.tell()
        file.file.seek(0)  # 重置到开头
        
        if file_size > cls.MAX_FILE_SIZE:
            raise HTTPException(
                status_code=status.HTTP_413_REQUEST_ENTITY_TOO_LARGE,
                detail=f"文件大小超过限制 ({cls.MAX_FILE_SIZE / 1024 / 1024}MB)"
            )
        
        # 检查文件类型 (magic number)
        file_content = file.file.read(2048)
        file.file.seek(0)
        
        detected_type = magic.from_buffer(file_content, mime=True)
        
        if detected_type not in allowed:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail=f"不支持的文件类型: {detected_type}"
            )
        
        # 检查文件扩展名
        file_ext = os.path.splitext(file.filename)[1].lower()
        expected_ext = allowed.get(detected_type)
        
        if expected_ext and file_ext != expected_ext:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail=f"文件扩展名与内容不匹配"
            )
        
        # 检查文件名中的路径遍历
        if '..' in file.filename or '/' in file.filename or '\\' in file.filename:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="文件名包含非法字符"
            )
        
        return {
            "filename": file.filename,
            "size": file_size,
            "type": detected_type,
            "is_valid": True
        }
    
    @classmethod
    def sanitize_filename(cls, filename: str) -> str:
        # 移除危险字符
        dangerous_chars = '<>:"/\\|?*'
        for char in dangerous_chars:
            filename = filename.replace(char, '_')
        
        # 限制长度
        name, ext = os.path.splitext(filename)
        name = name[:100]  # 限制主文件名长度
        
        return f"{name}{ext}"
```

### 4.1 备份服务 (backend/app/services/backup_service.py)

```python
from typing import Dict, List, Optional
from datetime import datetime, timedelta
import os
import shutil
import subprocess
from sqlalchemy import text

class BackupService:
    def __init__(self, db, backup_dir: str = "./backups"):
        self.db = db
        self.backup_dir = backup_dir
        os.makedirs(backup_dir, exist_ok=True)
    
    def create_database_backup(self, backup_name: Optional[str] = None) -> Dict:
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_name = backup_name or f"db_backup_{timestamp}"
        backup_path = os.path.join(self.backup_dir, f"{backup_name}.sql")
        
        # 使用 pg_dump 备份 PostgreSQL
        try:
            result = subprocess.run(
                ["pg_dump", "-Fc", "-f", backup_path, "xr_test_db"],
                capture_output=True,
                text=True,
                check=True
            )
            
            return {
                "success": True,
                "backup_name": backup_name,
                "backup_path": backup_path,
                "size_bytes": os.path.getsize(backup_path),
                "created_at": datetime.now().isoformat()
            }
        except subprocess.CalledProcessError as e:
            return {
                "success": False,
                "error": e.stderr,
                "backup_name": backup_name
            }
    
    def create_files_backup(self, source_dirs: List[str], 
                           backup_name: Optional[str] = None) -> Dict:
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_name = backup_name or f"files_backup_{timestamp}"
        backup_path = os.path.join(self.backup_dir, f"{backup_name}.tar.gz")
        
        try:
            # 创建压缩备份
            import tarfile
            with tarfile.open(backup_path, "w:gz") as tar:
                for source_dir in source_dirs:
                    if os.path.exists(source_dir):
                        tar.add(source_dir, arcname=os.path.basename(source_dir))
            
            return {
                "success": True,
                "backup_name": backup_name,
                "backup_path": backup_path,
                "size_bytes": os.path.getsize(backup_path),
                "created_at": datetime.now().isoformat()
            }
        except Exception as e:
            return {
                "success": False,
                "error": str(e),
                "backup_name": backup_name
            }
    
    def restore_database(self, backup_path: str) -> Dict:
        try:
            result = subprocess.run(
                ["pg_restore", "-d", "xr_test_db", "-c", backup_path],
                capture_output=True,
                text=True,
                check=True
            )
            
            return {
                "success": True,
                "restored_from": backup_path,
                "restored_at": datetime.now().isoformat()
            }
        except subprocess.CalledProcessError as e:
            return {
                "success": False,
                "error": e.stderr,
                "backup_path": backup_path
            }
    
    def list_backups(self) -> List[Dict]:
        backups = []
        
        for filename in os.listdir(self.backup_dir):
            filepath = os.path.join(self.backup_dir, filename)
            if os.path.isfile(filepath):
                stat = os.stat(filepath)
                backups.append({
                    "filename": filename,
                    "size_bytes": stat.st_size,
                    "created_at": datetime.fromtimestamp(stat.st_mtime).isoformat(),
                    "path": filepath
                })
        
        return sorted(backups, key=lambda x: x["created_at"], reverse=True)
    
    def cleanup_old_backups(self, retention_days: int = 30) -> Dict:
        cutoff = datetime.now() - timedelta(days=retention_days)
        removed = []
        
        for filename in os.listdir(self.backup_dir):
            filepath = os.path.join(self.backup_dir, filename)
            if os.path.isfile(filepath):
                mtime = datetime.fromtimestamp(os.stat(filepath).st_mtime)
                if mtime < cutoff:
                    os.remove(filepath)
                    removed.append(filename)
        
        return {
            "removed_count": len(removed),
            "removed_files": removed,
            "retention_days": retention_days
        }
    
    def verify_backup(self, backup_path: str) -> Dict:
        if not os.path.exists(backup_path):
            return {"valid": False, "error": "备份文件不存在"}
        
        # 检查文件完整性
        file_size = os.path.getsize(backup_path)
        if file_size == 0:
            return {"valid": False, "error": "备份文件为空"}
        
        # 对于数据库备份，尝试验证格式
        if backup_path.endswith('.sql'):
            try:
                with open(backup_path, 'r') as f:
                    first_line = f.readline()
                    if 'PostgreSQL' in first_line or 'pg_dump' in first_line:
                        return {"valid": True, "size_bytes": file_size}
            except:
                pass
        
        return {"valid": True, "size_bytes": file_size, "note": "基本检查通过"}
```
