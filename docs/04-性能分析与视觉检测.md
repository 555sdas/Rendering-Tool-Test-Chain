# 阶段四：性能分析与视觉检测

## 1. 目标

实现XR测试工具链的核心分析能力，包括：
- 渲染性能稳定性分析（平均FPS、P95/P99帧时间、掉帧率、长帧次数等）
- 视觉质量与画面缺陷检测（清晰度、色彩、闪烁、抖动、拖影等）
- 图像差异与参考帧对比（SSIM、PSNR、ΔE）
- 资源复杂度与部署负载分析（面数、Draw Call、Batch、Shader、材质、贴图等）

## 2. 性能稳定性分析服务

```python
from sqlalchemy import Column, Integer, Float, String, DateTime, JSON, ForeignKey, Enum
from sqlalchemy.sql import func
from app.database import Base
import enum

class AnalysisStatus(str, enum.Enum):
    PENDING = "pending"
    PROCESSING = "processing"
    COMPLETED = "completed"
    FAILED = "failed"

class PerformanceAnalysis(Base):
    __tablename__ = "performance_analyses"
    
    id = Column(Integer, primary_key=True, index=True)
    session_id = Column(Integer, ForeignKey("test_sessions.id"), nullable=False)
    status = Column(Enum(AnalysisStatus), default=AnalysisStatus.PENDING)
    
    # 帧率统计
    avg_fps = Column(Float)
    min_fps = Column(Float)
    max_fps = Column(Float)
    p95_fps = Column(Float)
    p99_fps = Column(Float)
    
    # 帧时间统计 (ms)
    avg_frame_time = Column(Float)
    min_frame_time = Column(Float)
    max_frame_time = Column(Float)
    p95_frame_time = Column(Float)
    p99_frame_time = Column(Float)
    
    # 掉帧分析
    dropped_frame_count = Column(Integer)
    dropped_frame_rate = Column(Float)  # 百分比
    long_frame_count = Column(Integer)  # 超过33ms的帧数
    long_frame_rate = Column(Float)
    
    # 稳定性
    fps_variance = Column(Float)
    fps_std_dev = Column(Float)
    
    # 场景切换
    scene_load_times = Column(JSON)  # [{"scene": "name", "duration_ms": 1234}]
    
    # 连续运行
    total_duration_seconds = Column(Float)
    crash_count = Column(Integer, default=0)
    freeze_count = Column(Integer, default=0)
    
    # 资源使用峰值
    peak_cpu_percent = Column(Float)
    peak_gpu_percent = Column(Float)
    peak_memory_mb = Column(Float)
    peak_vram_mb = Column(Float)
    
    # 分析结果
    issues = Column(JSON)  # [{"type": "dropped_frames", "severity": "high", "description": "..."}]
    recommendations = Column(JSON)
    
    created_at = Column(DateTime(timezone=True), server_default=func.now())
    completed_at = Column(DateTime(timezone=True))
```

### 2.2 性能分析服务 (backend/app/services/performance_analysis_service.py)

```python
from sqlalchemy.orm import Session
from sqlalchemy import func
from typing import List, Dict, Any, Optional
from datetime import datetime
import numpy as np
from scipy import stats

from app.models.performance_analysis import PerformanceAnalysis, AnalysisStatus
from app.models.performance_sample import PerformanceSample
from app.models.test_session import TestSession

class PerformanceAnalysisService:
    def __init__(self, db: Session):
        self.db = db
    
    def analyze_session(self, session_id: int) -> PerformanceAnalysis:
        session = self.db.query(TestSession).filter(TestSession.id == session_id).first()
        if not session:
            raise ValueError(f"Session {session_id} not found")
        
        # 获取所有采样数据
        samples = self.db.query(PerformanceSample).filter(
            PerformanceSample.session_id == session_id
        ).order_by(PerformanceSample.timestamp).all()
        
        if not samples:
            raise ValueError(f"No samples found for session {session_id}")
        
        analysis = PerformanceAnalysis(
            session_id=session_id,
            status=AnalysisStatus.PROCESSING
        )
        self.db.add(analysis)
        self.db.commit()
        
        try:
            # 提取数据数组
            fps_values = np.array([s.fps for s in samples if s.fps is not None])
            frame_times = np.array([s.delta_time_ms for s in samples if s.delta_time_ms is not None])
            cpu_values = np.array([s.cpu_usage_percent for s in samples if s.cpu_usage_percent is not None])
            gpu_values = np.array([s.gpu_usage_percent for s in samples if s.gpu_usage_percent is not None])
            memory_values = np.array([s.memory_usage_bytes for s in samples if s.memory_usage_bytes is not None])
            vram_values = np.array([s.vram_usage_bytes for s in samples if s.vram_usage_bytes is not None])
            
            # 帧率统计
            if len(fps_values) > 0:
                analysis.avg_fps = float(np.mean(fps_values))
                analysis.min_fps = float(np.min(fps_values))
                analysis.max_fps = float(np.max(fps_values))
                analysis.p95_fps = float(np.percentile(fps_values, 95))
                analysis.p99_fps = float(np.percentile(fps_values, 99))
                analysis.fps_variance = float(np.var(fps_values))
                analysis.fps_std_dev = float(np.std(fps_values))
            
            # 帧时间统计
            if len(frame_times) > 0:
                analysis.avg_frame_time = float(np.mean(frame_times))
                analysis.min_frame_time = float(np.min(frame_times))
                analysis.max_frame_time = float(np.max(frame_times))
                analysis.p95_frame_time = float(np.percentile(frame_times, 95))
                analysis.p99_frame_time = float(np.percentile(frame_times, 99))
                
                # 掉帧分析 (假设目标帧率为90fps = 11.1ms, 或72fps = 13.9ms)
                target_frame_time = 11.1  # 默认90fps
                dropped_frames = frame_times[frame_times > target_frame_time * 1.5]
                long_frames = frame_times[frame_times > 33.3]  # 超过30fps的帧时间
                
                analysis.dropped_frame_count = int(len(dropped_frames))
                analysis.dropped_frame_rate = float(len(dropped_frames) / len(frame_times) * 100)
                analysis.long_frame_count = int(len(long_frames))
                analysis.long_frame_rate = float(len(long_frames) / len(frame_times) * 100)
            
            # 资源使用峰值
            if len(cpu_values) > 0:
                analysis.peak_cpu_percent = float(np.max(cpu_values))
            if len(gpu_values) > 0:
                analysis.peak_gpu_percent = float(np.max(gpu_values))
            if len(memory_values) > 0:
                analysis.peak_memory_mb = float(np.max(memory_values) / (1024 * 1024))
            if len(vram_values) > 0:
                analysis.peak_vram_mb = float(np.max(vram_values) / (1024 * 1024))
            
            # 总时长
            if session.start_time and session.end_time:
                analysis.total_duration_seconds = (session.end_time - session.start_time).total_seconds()
            
            # 生成问题和建议
            analysis.issues = self._detect_issues(analysis, fps_values, frame_times)
            analysis.recommendations = self._generate_recommendations(analysis)
            
            analysis.status = AnalysisStatus.COMPLETED
            analysis.completed_at = datetime.utcnow()
            
        except Exception as e:
            analysis.status = AnalysisStatus.FAILED
            analysis.issues = [{"type": "analysis_error", "severity": "critical", "description": str(e)}]
        
        self.db.commit()
        self.db.refresh(analysis)
        return analysis
    
    def _detect_issues(self, analysis: PerformanceAnalysis, fps_values: np.ndarray, 
                       frame_times: np.ndarray) -> List[Dict]:
        issues = []
        
        # 帧率问题
        if analysis.avg_fps and analysis.avg_fps < 72:
            issues.append({
                "type": "low_fps",
                "severity": "high" if analysis.avg_fps < 60 else "medium",
                "description": f"平均帧率 {analysis.avg_fps:.1f} FPS 低于目标 72 FPS",
                "metric": "avg_fps",
                "value": analysis.avg_fps,
                "threshold": 72
            })
        
        # 掉帧问题
        if analysis.dropped_frame_rate and analysis.dropped_frame_rate > 5:
            issues.append({
                "type": "dropped_frames",
                "severity": "high" if analysis.dropped_frame_rate > 10 else "medium",
                "description": f"掉帧率 {analysis.dropped_frame_rate:.1f}% 过高",
                "metric": "dropped_frame_rate",
                "value": analysis.dropped_frame_rate,
                "threshold": 5
            })
        
        # 长帧问题
        if analysis.long_frame_count and analysis.long_frame_count > 10:
            issues.append({
                "type": "long_frames",
                "severity": "medium",
                "description": f"检测到 {analysis.long_frame_count} 次长帧(>33ms)",
                "metric": "long_frame_count",
                "value": analysis.long_frame_count,
                "threshold": 10
            })
        
        # 帧时间波动
        if analysis.p95_frame_time and analysis.p95_frame_time > 20:
            issues.append({
                "type": "frame_time_variance",
                "severity": "medium",
                "description": f"P95帧时间 {analysis.p95_frame_time:.1f}ms 波动较大",
                "metric": "p95_frame_time",
                "value": analysis.p95_frame_time,
                "threshold": 20
            })
        
        # CPU问题
        if analysis.peak_cpu_percent and analysis.peak_cpu_percent > 90:
            issues.append({
                "type": "high_cpu_usage",
                "severity": "high",
                "description": f"CPU峰值使用率 {analysis.peak_cpu_percent:.1f}%",
                "metric": "peak_cpu_percent",
                "value": analysis.peak_cpu_percent,
                "threshold": 90
            })
        
        # GPU问题
        if analysis.peak_gpu_percent and analysis.peak_gpu_percent > 90:
            issues.append({
                "type": "high_gpu_usage",
                "severity": "high",
                "description": f"GPU峰值使用率 {analysis.peak_gpu_percent:.1f}%",
                "metric": "peak_gpu_percent",
                "value": analysis.peak_gpu_percent,
                "threshold": 90
            })
        
        # 内存问题
        if analysis.peak_memory_mb and analysis.peak_memory_mb > 2048:
            issues.append({
                "type": "high_memory_usage",
                "severity": "medium",
                "description": f"内存峰值 {analysis.peak_memory_mb:.0f}MB",
                "metric": "peak_memory_mb",
                "value": analysis.peak_memory_mb,
                "threshold": 2048
            })
        
        return issues
    
    def _generate_recommendations(self, analysis: PerformanceAnalysis) -> List[Dict]:
        recommendations = []
        
        if analysis.avg_fps and analysis.avg_fps < 72:
            recommendations.append({
                "category": "performance",
                "priority": "high",
                "suggestion": "优化渲染管线，考虑降低渲染分辨率或简化场景复杂度",
                "details": "检查Draw Call数量、Shader复杂度和纹理分辨率"
            })
        
        if analysis.dropped_frame_rate and analysis.dropped_frame_rate > 5:
            recommendations.append({
                "category": "stability",
                "priority": "high",
                "suggestion": "分析掉帧时刻的CPU/GPU负载，排查GC或资源加载导致的卡顿",
                "details": "使用Unity Profiler分析具体瓶颈"
            })
        
        if analysis.peak_cpu_percent and analysis.peak_cpu_percent > 80:
            recommendations.append({
                "category": "cpu_optimization",
                "priority": "medium",
                "suggestion": "优化CPU密集型操作，考虑使用Job System或Burst Compiler",
                "details": "检查物理计算、AI逻辑和动画更新"
            })
        
        if analysis.peak_gpu_percent and analysis.peak_gpu_percent > 80:
            recommendations.append({
                "category": "gpu_optimization",
                "priority": "medium",
                "suggestion": "优化GPU负载，考虑使用GPU Instancing、LOD或遮挡剔除",
                "details": "检查过度绘制、复杂Shader和全屏后处理效果"
            })
        
        if analysis.peak_memory_mb and analysis.peak_memory_mb > 1500:
            recommendations.append({
                "category": "memory_optimization",
                "priority": "medium",
                "suggestion": "优化内存使用，及时释放不用的资源，使用Addressables管理资源",
                "details": "检查纹理内存、Mesh内存和音频内存"
            })
        
        return recommendations
    
    def compare_sessions(self, session_ids: List[int]) -> Dict:
        analyses = self.db.query(PerformanceAnalysis).filter(
            PerformanceAnalysis.session_id.in_(session_ids),
            PerformanceAnalysis.status == AnalysisStatus.COMPLETED
        ).all()
        
        if len(analyses) < 2:
            raise ValueError("Need at least 2 completed analyses for comparison")
        
        comparison = {
            "sessions": [],
            "metrics_comparison": {},
            "winner": None
        }
        
        for analysis in analyses:
            session = self.db.query(TestSession).filter(TestSession.id == analysis.session_id).first()
            comparison["sessions"].append({
                "session_id": analysis.session_id,
                "device": session.device_info.get("deviceModel") if session else "Unknown",
                "avg_fps": analysis.avg_fps,
                "p95_frame_time": analysis.p95_frame_time,
                "dropped_frame_rate": analysis.dropped_frame_rate,
                "peak_memory_mb": analysis.peak_memory_mb
            })
        
        # 找出综合表现最好的会话
        scores = []
        for s in comparison["sessions"]:
            score = 0
            if s["avg_fps"]:
                score += s["avg_fps"] * 10
            if s["dropped_frame_rate"] is not None:
                score -= s["dropped_frame_rate"] * 5
            if s["peak_memory_mb"]:
                score -= s["peak_memory_mb"] * 0.01
            scores.append(score)
        
        best_idx = scores.index(max(scores))
        comparison["winner"] = comparison["sessions"][best_idx]["session_id"]
        
        return comparison
```

### 3.1 视觉检测服务 (backend/app/services/visual_detection_service.py)

```python
import cv2
import numpy as np
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass
from enum import Enum

class DefectType(str, Enum):
    BLUR = "blur"
    FLICKER = "flicker"
    COLOR_ABNORMAL = "color_abnormal"
    BRIGHTNESS_ISSUE = "brightness_issue"
    ALIASING = "aliasing"
    TEXTURE_FILTERING = "texture_filtering"
    LEFT_RIGHT_MISMATCH = "left_right_mismatch"
    GHOSTING = "ghosting"
    TEARING = "tearing"

@dataclass
class DefectDetection:
    defect_type: DefectType
    severity: str  # low, medium, high, critical
    confidence: float
    description: str
    bounding_box: Optional[Tuple[int, int, int, int]] = None
    frame_index: Optional[int] = None
    timestamp: Optional[float] = None

class VisualDetectionService:
    def __init__(self):
        self.blur_threshold = 100.0  # Laplacian方差阈值
        self.flicker_threshold = 50.0  # 帧间差异阈值
        self.brightness_low = 30
        self.brightness_high = 220
    
    def analyze_frame(self, frame: np.ndarray, prev_frame: Optional[np.ndarray] = None) -> List[DefectDetection]:
        defects = []
        
        # 清晰度检测
        blur_defect = self._detect_blur(frame)
        if blur_defect:
            defects.append(blur_defect)
        
        # 闪烁检测
        if prev_frame is not None:
            flicker_defect = self._detect_flicker(frame, prev_frame)
            if flicker_defect:
                defects.append(flicker_defect)
        
        # 色彩/亮度检测
        color_defects = self._detect_color_issues(frame)
        defects.extend(color_defects)
        
        # 抗锯齿检测
        aliasing_defect = self._detect_aliasing(frame)
        if aliasing_defect:
            defects.append(aliasing_defect)
        
        return defects
    
    def _detect_blur(self, frame: np.ndarray) -> Optional[DefectDetection]:
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        laplacian_var = cv2.Laplacian(gray, cv2.CV_64F).var()
        
        if laplacian_var < self.blur_threshold:
            severity = "high" if laplacian_var < 50 else "medium"
            return DefectDetection(
                defect_type=DefectType.BLUR,
                severity=severity,
                confidence=min(1.0, (self.blur_threshold - laplacian_var) / self.blur_threshold),
                description=f"画面清晰度不足 (Laplacian方差: {laplacian_var:.1f})"
            )
        return None
    
    def _detect_flicker(self, current: np.ndarray, previous: np.ndarray) -> Optional[DefectDetection]:
        if current.shape != previous.shape:
            return None
        
        diff = cv2.absdiff(current, previous)
        mean_diff = np.mean(diff)
        
        if mean_diff > self.flicker_threshold:
            severity = "high" if mean_diff > 100 else "medium"
            return DefectDetection(
                defect_type=DefectType.FLICKER,
                severity=severity,
                confidence=min(1.0, mean_diff / 200.0),
                description=f"检测到画面闪烁 (帧间差异: {mean_diff:.1f})"
            )
        return None
    
    def _detect_color_issues(self, frame: np.ndarray) -> List[DefectDetection]:
        defects = []
        
        # 亮度分析
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        mean_brightness = np.mean(gray)
        
        if mean_brightness < self.brightness_low:
            defects.append(DefectDetection(
                defect_type=DefectType.BRIGHTNESS_ISSUE,
                severity="medium",
                confidence=min(1.0, (self.brightness_low - mean_brightness) / self.brightness_low),
                description=f"画面过暗 (平均亮度: {mean_brightness:.1f})"
            ))
        elif mean_brightness > self.brightness_high:
            defects.append(DefectDetection(
                defect_type=DefectType.BRIGHTNESS_ISSUE,
                severity="medium",
                confidence=min(1.0, (mean_brightness - self.brightness_high) / (255 - self.brightness_high)),
                description=f"画面过曝 (平均亮度: {mean_brightness:.1f})"
            ))
        
        # 色彩饱和度分析
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        saturation = hsv[:, :, 1]
        mean_saturation = np.mean(saturation)
        
        if mean_saturation < 20:
            defects.append(DefectDetection(
                defect_type=DefectType.COLOR_ABNORMAL,
                severity="low",
                confidence=0.5,
                description=f"画面色彩饱和度偏低 (平均饱和度: {mean_saturation:.1f})"
            ))
        
        return defects
    
    def _detect_aliasing(self, frame: np.ndarray) -> Optional[DefectDetection]:
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        # 使用Sobel算子检测边缘
        sobelx = cv2.Sobel(gray, cv2.CV_64F, 1, 0, ksize=3)
        sobely = cv2.Sobel(gray, cv2.CV_64F, 0, 1, ksize=3)
        sobel_mag = np.sqrt(sobelx**2 + sobely**2)
        
        # 检测锯齿状边缘（高频噪声）
        edges = cv2.Canny(gray, 50, 150)
        edge_density = np.sum(edges > 0) / (edges.shape[0] * edges.shape[1])
        
        if edge_density > 0.1:
            return DefectDetection(
                defect_type=DefectType.ALIASING,
                severity="low",
                confidence=min(1.0, edge_density * 2),
                description=f"检测到较多锯齿边缘 (边缘密度: {edge_density:.3f})"
            )
        return None
    
    def detect_left_right_mismatch(self, left_frame: np.ndarray, right_frame: np.ndarray) -> Optional[DefectDetection]:
        if left_frame.shape != right_frame.shape:
            return DefectDetection(
                defect_type=DefectType.LEFT_RIGHT_MISMATCH,
                severity="critical",
                confidence=1.0,
                description="左右眼画面尺寸不一致"
            )
        
        diff = cv2.absdiff(left_frame, right_frame)
        mean_diff = np.mean(diff)
        
        # 左右眼应有合理差异（视差），但不应过大
        if mean_diff > 100:
            return DefectDetection(
                defect_type=DefectType.LEFT_RIGHT_MISMATCH,
                severity="high",
                confidence=min(1.0, mean_diff / 255.0),
                description=f"左右眼画面差异过大 (平均差异: {mean_diff:.1f})"
            )
        
        return None
    
    def analyze_video_stream(self, frames: List[np.ndarray], timestamps: List[float]) -> Dict:
        all_defects = []
        prev_frame = None
        
        for i, frame in enumerate(frames):
            frame_defects = self.analyze_frame(frame, prev_frame)
            for defect in frame_defects:
                defect.frame_index = i
                defect.timestamp = timestamps[i] if i < len(timestamps) else None
            all_defects.extend(frame_defects)
            prev_frame = frame.copy()
        
        # 统计
        defect_counts = {}
        severity_counts = {"low": 0, "medium": 0, "high": 0, "critical": 0}
        
        for defect in all_defects:
            defect_type = defect.defect_type.value
            defect_counts[defect_type] = defect_counts.get(defect_type, 0) + 1
            severity_counts[defect.severity] = severity_counts.get(defect.severity, 0) + 1
        
        return {
            "total_frames": len(frames),
            "total_defects": len(all_defects),
            "defect_counts": defect_counts,
            "severity_distribution": severity_counts,
            "defects": [
                {
                    "type": d.defect_type.value,
                    "severity": d.severity,
                    "confidence": d.confidence,
                    "description": d.description,
                    "frame_index": d.frame_index,
                    "timestamp": d.timestamp
                }
                for d in all_defects
            ]
        }
```

```python
import cv2
import numpy as np
from typing import Dict, Tuple, Optional
from skimage.metrics import structural_similarity as ssim
import sewar

class ImageComparisonService:
    def __init__(self):
        self.ssim_threshold = 0.95
        self.psnr_threshold = 30.0
        self.delta_e_threshold = 5.0
    
    def compare_images(self, reference: np.ndarray, target: np.ndarray) -> Dict:
        if reference.shape != target.shape:
            target = cv2.resize(target, (reference.shape[1], reference.shape[0]))
        
        results = {
            "dimensions": {
                "width": reference.shape[1],
                "height": reference.shape[0]
            },
            "metrics": {},
            "difference_map": None,
            "passed": True
        }
        
        # SSIM
        try:
            ssim_score, ssim_map = self._calculate_ssim(reference, target)
            results["metrics"]["ssim"] = {
                "value": round(ssim_score, 4),
                "threshold": self.ssim_threshold,
                "passed": ssim_score >= self.ssim_threshold
            }
            results["passed"] = results["passed"] and results["metrics"]["ssim"]["passed"]
        except Exception as e:
            results["metrics"]["ssim"] = {"error": str(e)}
        
        # PSNR
        try:
            psnr_score = self._calculate_psnr(reference, target)
            results["metrics"]["psnr"] = {
                "value": round(psnr_score, 2),
                "threshold": self.psnr_threshold,
                "passed": psnr_score >= self.psnr_threshold
            }
            results["passed"] = results["passed"] and results["metrics"]["psnr"]["passed"]
        except Exception as e:
            results["metrics"]["psnr"] = {"error": str(e)}
        
        # Delta E (色彩差异)
        try:
            delta_e_score, delta_e_map = self._calculate_delta_e(reference, target)
            results["metrics"]["delta_e"] = {
                "value": round(delta_e_score, 2),
                "threshold": self.delta_e_threshold,
                "passed": delta_e_score <= self.delta_e_threshold
            }
            results["passed"] = results["passed"] and results["metrics"]["delta_e"]["passed"]
        except Exception as e:
            results["metrics"]["delta_e"] = {"error": str(e)}
        
        # 生成差异图
        try:
            diff_map = self._generate_difference_map(reference, target)
            results["difference_map"] = diff_map
        except Exception as e:
            results["difference_map_error"] = str(e)
        
        return results
    
    def _calculate_ssim(self, ref: np.ndarray, target: np.ndarray) -> Tuple[float, np.ndarray]:
        gray_ref = cv2.cvtColor(ref, cv2.COLOR_BGR2GRAY)
        gray_target = cv2.cvtColor(target, cv2.COLOR_BGR2GRAY)
        
        score, diff = ssim(gray_ref, gray_target, full=True)
        return score, diff
    
    def _calculate_psnr(self, ref: np.ndarray, target: np.ndarray) -> float:
        mse = np.mean((ref.astype(np.float64) - target.astype(np.float64)) ** 2)
        if mse == 0:
            return float('inf')
        
        max_pixel = 255.0
        psnr = 20 * np.log10(max_pixel / np.sqrt(mse))
        return psnr
    
    def _calculate_delta_e(self, ref: np.ndarray, target: np.ndarray) -> Tuple[float, np.ndarray]:
        # 转换到Lab色彩空间
        lab_ref = cv2.cvtColor(ref, cv2.COLOR_BGR2Lab).astype(np.float32)
        lab_target = cv2.cvtColor(target, cv2.COLOR_BGR2Lab).astype(np.float32)
        
        # CIE76 Delta E
        delta_e = np.sqrt(np.sum((lab_ref - lab_target) ** 2, axis=2))
        mean_delta_e = np.mean(delta_e)
        
        return mean_delta_e, delta_e
    
    def _generate_difference_map(self, ref: np.ndarray, target: np.ndarray) -> Dict:
        # 像素级差异
        diff = cv2.absdiff(ref, target)
        
        # 增强差异可视化
        diff_gray = cv2.cvtColor(diff, cv2.COLOR_BGR2GRAY)
        _, diff_thresh = cv2.threshold(diff_gray, 30, 255, cv2.THRESH_BINARY)
        
        # 热力图
        diff_heatmap = cv2.applyColorMap(diff_gray, cv2.COLORMAP_JET)
        
        # 叠加图
        overlay = cv2.addWeighted(ref, 0.7, diff_heatmap, 0.3, 0)
        
        return {
            "pixel_diff": diff,
            "threshold_mask": diff_thresh,
            "heatmap": diff_heatmap,
            "overlay": overlay
        }
    
    def compare_with_reference_frame(self, session_id: int, reference_frame_path: str, 
                                     target_frame_paths: List[str]) -> Dict:
        reference = cv2.imread(reference_frame_path)
        if reference is None:
            raise ValueError(f"Cannot load reference frame: {reference_frame_path}")
        
        comparisons = []
        for target_path in target_frame_paths:
            target = cv2.imread(target_path)
            if target is None:
                continue
            
            result = self.compare_images(reference, target)
            result["target_frame"] = target_path
            comparisons.append(result)
        
        # 统计
        pass_count = sum(1 for c in comparisons if c["passed"])
        
        return {
            "reference_frame": reference_frame_path,
            "total_comparisons": len(comparisons),
            "passed": pass_count,
            "failed": len(comparisons) - pass_count,
            "comparisons": comparisons
        }
```

```python
from typing import Dict, List, Any, Optional
import json

class ResourceAnalysisService:
    def __init__(self):
        self.complexity_levels = {
            "low": {"triangles": 50000, "draw_calls": 100, "materials": 20},
            "medium": {"triangles": 200000, "draw_calls": 300, "materials": 50},

    def analyze_scene_resources(self, scene_data: Dict) -> Dict:
        analysis = {
            "scene_name": scene_data.get("name", "Unknown"),
            "complexity_level": "unknown",
            "metrics": {},
            "issues": [],
            "recommendations": []
        }
        
        # 模型统计
        meshes = scene_data.get("meshes", [])
        total_triangles = sum(m.get("triangle_count", 0) for m in meshes)
        total_vertices = sum(m.get("vertex_count", 0) for m in meshes)
        
        analysis["metrics"]["geometry"] = {
            "mesh_count": len(meshes),
            "total_triangles": total_triangles,
            "total_vertices": total_vertices,
            "avg_triangles_per_mesh": total_triangles / len(meshes) if meshes else 0
        }
        
        # 渲染统计
        render_stats = scene_data.get("render_stats", {})

            "batches": render_stats.get("batches", 0),
            "shadow_casters": render_stats.get("shadow_casters", 0)
        }
        
        # Shader统计
        shaders = scene_data.get("shaders", [])
        analysis["metrics"]["shaders"] = {
            "total_count": len(shaders),
            "unique_count": len(set(s.get("name") for s in shaders)),
            "complex_shaders": sum(1 for s in shaders if s.get("is_complex", False)),
            "variant_count": sum(s.get("variant_count", 0) for s in shaders)
        }
        
        # 材质统计
        materials = scene_data.get("materials", [])
        analysis["metrics"]["materials"] = {
            "total_count": len(materials),
            "unique_count": len(set(m.get("name") for m in materials)),
            "transparent_count": sum(1 for m in materials if m.get("is_transparent", False)),
            "custom_shader_count": sum(1 for m in materials if m.get("uses_custom_shader", False))
        }
        
        # 贴图/纹理统计
        textures = scene_data.get("textures", [])
        total_texture_memory = sum(t.get("memory_bytes", 0) for t in textures)
        
        analysis["metrics"]["textures"] = {
            "total_count": len(textures),
            "total_memory_mb": round(total_texture_memory / (1024 * 1024), 2),
            "max_resolution": max((t.get("width", 0) for t in textures), default=0),
            "compressed_count": sum(1 for t in textures if t.get("is_compressed", False))
        }
        
        # 包体大小
        analysis["metrics"]["build"] = {
            "total_size_mb": round(scene_data.get("build_size_bytes", 0) / (1024 * 1024), 2),
            "scene_size_mb": round(scene_data.get("scene_size_bytes", 0) / (1024 * 1024), 2),
            "asset_bundle_size_mb": round(scene_data.get("asset_bundle_size_bytes", 0) / (1024 * 1024), 2)
        }
        
        # 加载时间
        analysis["metrics"]["loading"] = {
            "scene_load_time_ms": scene_data.get("scene_load_time_ms", 0),
            "asset_load_time_ms": scene_data.get("asset_load_time_ms", 0),
            "shader_compile_time_ms": scene_data.get("shader_compile_time_ms", 0)
        }
        
        # 复杂度评估
        analysis["complexity_level"] = self._assess_complexity(analysis["metrics"])
        
        # 问题检测
        analysis["issues"] = self._detect_resource_issues(analysis["metrics"])
        
        # 优化建议
        analysis["recommendations"] = self._generate_resource_recommendations(analysis["metrics"], analysis["issues"])
        
        return analysis
    
    def _assess_complexity(self, metrics: Dict) -> str:
        triangles = metrics["geometry"]["total_triangles"]
        draw_calls = metrics["rendering"]["draw_calls"]
        materials = metrics["materials"]["total_count"]
        
        if (triangles > self.complexity_levels["high"]["triangles"] or
            draw_calls > self.complexity_levels["high"]["draw_calls"] or
            materials > self.complexity_levels["high"]["materials"]):
            return "high"
        elif (triangles > self.complexity_levels["medium"]["triangles"] or
              draw_calls > self.complexity_levels["medium"]["draw_calls"] or
              materials > self.complexity_levels["medium"]["materials"]):
            return "medium"
        else:
            return "low"
    
    def _detect_resource_issues(self, metrics: Dict) -> List[Dict]:
        issues = []
        
        # 面数过多
        triangles = metrics["geometry"]["total_triangles"]
        if triangles > 500000:

                "type": "high_triangle_count",
                "severity": "high",
                "description": f"场景总面数 {triangles:,} 过高，建议控制在 50万 以下",
                "metric": "total_triangles",
                "value": triangles,
                "threshold": 500000
            })
        
        # Draw Call过多
        draw_calls = metrics["rendering"]["draw_calls"]
        if draw_calls > 500:
            issues.append({
                "type": "high_draw_calls",
                "severity": "high",
                "description": f"Draw Call数量 {draw_calls} 过高，建议控制在 500 以下",
                "metric": "draw_calls",
                "value": draw_calls,
                "threshold": 500
            })
        
        # 贴图内存过大
        texture_memory = metrics["textures"]["total_memory_mb"]
        if texture_memory > 512:
            issues.append({
                "type": "high_texture_memory",
                "severity": "medium",
                "description": f"贴图内存占用 {texture_memory:.1f}MB 过大",
                "metric": "texture_memory_mb",
                "value": texture_memory,
                "threshold": 512
            })
        
        # 加载时间过长
        load_time = metrics["loading"]["scene_load_time_ms"]
        if load_time > 5000:
            issues.append({
                "type": "long_load_time",
                "severity": "medium",
                "description": f"场景加载时间 {load_time/1000:.1f}s 过长",
                "metric": "scene_load_time_ms",
                "value": load_time,
                "threshold": 5000
            })
        
        # Shader变体过多
        shader_variants = metrics["shaders"]["variant_count"]
        if shader_variants > 1000:
            issues.append({
                "type": "high_shader_variants",
                "severity": "medium",
                "description": f"Shader变体数量 {shader_variants} 过多，影响编译时间和内存",
                "metric": "shader_variant_count",
                "value": shader_variants,
                "threshold": 1000
            })
        
        return issues
    
    def _generate_resource_recommendations(self, metrics: Dict, issues: List[Dict]) -> List[Dict]:
        recommendations = []
```
